43LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam){
44 UINT nMenuId = 0; //menu identifier for menu command messages
45 static bool bResizing = false;
46
47 switch(message){
48 case WM_CREATE: //window has been created
49 g_pMain = new CMain(hWnd); //create the main class
50 g_pMain->CreateBitmap(g_nWidth, g_nHeight);
51 return 0;
52
53 case WM_DESTROY: //window has been removed from the screen
54 delete g_pMain; //delete the main class
55 PostQuitMessage(0); //ready to shut down
56 return 0;
57
58 case WM_PAINT: //window needs to be redrawn
59 g_pMain->OnPaint();
60 return 0;
61
62 case WM_COMMAND: //user has selected a command from the menu
63 nMenuId = LOWORD(wParam); //menu id
64
65 switch(nMenuId){
66 case IDM_GENERATE_PERLINNOISE:
67 g_pMain->GeneratePerlinNoise();
68 InvalidateRect(hWnd, nullptr, FALSE);
69 break;
70
71 case IDM_GENERATE_PIXELNOISE:
72 g_pMain->GeneratePixelNoise();
73 InvalidateRect(hWnd, nullptr, FALSE);
74 break;
75
76 case IDM_GENERATE_VALUENOISE:
77 g_pMain->GenerateValueNoise();
78 InvalidateRect(hWnd, nullptr, FALSE);
79 break;
80
81 case IDM_FILE_SAVE: //save bitmap to image file
82 SaveBitmap(hWnd, g_pMain->GetBitmap());
83 break;
84
85 case IDM_FILE_QUIT: //so long, farewell, auf weidersehn, goodbye!
86 SendMessage(hWnd, WM_CLOSE, 0, 0);
87 break;
88 } //switch
89
90 return 0; //all is good
91
92 default:
93 return DefWindowProc(hWnd, message, wParam, lParam); //not my message
94 } //switch
95} //WndProc

102void InitWindow(HINSTANCE hInst, INT nShow, WNDPROC WndProc){
103 const LPCWSTR appname = L"2DNoiseGenerator";
104
105 WNDCLASSEX wndClass = {0}; //extended window class structure
106
107 wndClass.cbSize = sizeof(WNDCLASSEX);
108 wndClass.style = CS_HREDRAW | CS_VREDRAW;
109 wndClass.lpfnWndProc = WndProc;
110 wndClass.cbClsExtra = 0;
111 wndClass.cbWndExtra = 0;
112 wndClass.hInstance = hInst;
113 wndClass.hIcon = LoadIcon(hInst, MAKEINTRESOURCE(IDI_ICON1));
114 wndClass.hCursor = LoadCursor(nullptr, IDC_ARROW);
115 wndClass.hbrBackground = nullptr;
116 wndClass.lpszMenuName = nullptr;
117 wndClass.lpszClassName = appname;
118
119 RegisterClassEx(&wndClass);
120
121 const DWORD dwStyle = WS_CAPTION | WS_MINIMIZEBOX | WS_SYSMENU;
122 const DWORD dwStyleEx = WS_EX_APPWINDOW | WS_EX_DLGMODALFRAME;
123
124 RECT r{0};
125 r.right = g_nWidth;
126 r.bottom = g_nHeight + GetSystemMetrics(SM_CYMENU);
127 AdjustWindowRectEx(&r, dwStyle, FALSE, dwStyleEx);
128
129 const HWND hwnd = CreateWindowEx(dwStyleEx, appname, appname, dwStyle,
130 CW_USEDEFAULT, CW_USEDEFAULT, r.right - r.left, r.bottom - r.top,
131 nullptr, nullptr, hInst, nullptr);
132
133 ShowWindow(hwnd, nShow);
134 UpdateWindow(hwnd);
135} //InitWindow
