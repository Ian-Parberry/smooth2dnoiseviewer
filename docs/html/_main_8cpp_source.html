239void InitWindow(HINSTANCE hInst, INT nShow, WNDPROC WndProc){
240 const LPCWSTR appname = L"2D Noise Generator";
241
242 WNDCLASSEX wndClass = {0}; //extended window class structure
243
244 wndClass.cbSize = sizeof(WNDCLASSEX);
245 wndClass.style = CS_HREDRAW | CS_VREDRAW;
246 wndClass.lpfnWndProc = WndProc;
247 wndClass.cbClsExtra = 0;
248 wndClass.cbWndExtra = 0;
249 wndClass.hInstance = hInst;
250 wndClass.hIcon = LoadIcon(hInst, MAKEINTRESOURCE(IDI_ICON1));
251 wndClass.hCursor = LoadCursor(nullptr, IDC_ARROW);
252 wndClass.hbrBackground = nullptr;
253 wndClass.lpszMenuName = nullptr;
254 wndClass.lpszClassName = appname;
255
256 RegisterClassEx(&wndClass);
257
258 const DWORD dwStyle = WS_CAPTION | WS_MINIMIZEBOX | WS_SYSMENU;
259 const DWORD dwStyleEx = WS_EX_APPWINDOW | WS_EX_DLGMODALFRAME;
260
261 RECT r{0};
262 r.right = g_nWidth;
263 r.bottom = g_nHeight + GetSystemMetrics(SM_CYMENU);
264 AdjustWindowRectEx(&r, dwStyle, FALSE, dwStyleEx);
265
266 const HWND hwnd = CreateWindowEx(dwStyleEx, appname, appname, dwStyle,
267 CW_USEDEFAULT, CW_USEDEFAULT, r.right - r.left, r.bottom - r.top,
268 nullptr, nullptr, hInst, nullptr);
269
270 ShowWindow(hwnd, nShow);
271 UpdateWindow(hwnd);
272} //InitWindow
