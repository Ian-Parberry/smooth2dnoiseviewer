80void CPerlinNoise2D::RandomizePermutation(){
81 for(size_t i=0; i<m_nSize; i++) //identity permutation
82 m_nPerm[i] = i;
83
84 for(size_t i=0; i<m_nSize; i++) //randomize
85 std::swap(m_nPerm[i], m_nPerm[i + rand()%(m_nSize - i)]);
86} //RandomizePermutation

106void CPerlinNoise2D::RandomizeTableMidpoint(size_t i, size_t j, float alpha){
107 std::uniform_real_distribution<float> d(-1.0f, 1.0f);
108
109 if(j > i + 1){ //base of recursion is "do nothing"
110 const size_t mid = (i + j)/2; //mid point
111
112 const float fMean = (m_fTable[i] + m_fTable[j - 1])/2.0f; //average of ends
113 const float fRand = alpha*d(m_stdRandom); //random offset
114 m_fTable[mid] = clamp(-1.0f, fMean + fRand, 1.0f); //mid point is average plus offset
115 alpha *= 0.5f; //increase lacunarity
116
117 RandomizeTableMidpoint(i, mid, alpha); //recurse on first half
118 RandomizeTableMidpoint(mid, j, alpha); //recurse on second half
119 } //if
120} //RandomizeTableMidpoint

151void CPerlinNoise2D::RandomizeTableNormal(){
152 std::normal_distribution<float> d(500.0f, 200.0f);

180void CPerlinNoise2D::RandomizeTableExp(){
181 std::exponential_distribution<float> d(8.0f);
182
183 const size_t half = m_nSize/2;
184
185 for(size_t i=0; i<half; i++) //positive values
186 m_fTable[i] = clamp(0.0f, d(m_stdRandom), 1.0f);
187
188 for(size_t i=half; i<m_nSize; i++) //negative values
189 m_fTable[i] = -clamp(0.0f, d(m_stdRandom), 1.0f);
190} //RandomizeTableExp

197void CPerlinNoise2D::RandomizeTable(eDistribution d){
198 m_stdRandom.seed(timeGetTime()); //reset PRNG
199
200 switch(d){
201 case eDistribution::Uniform: RandomizeTableUniform(); break;
202 case eDistribution::Cosine: RandomizeTableCos(); break;
203 case eDistribution::Normal: RandomizeTableNormal(); break;
204 case eDistribution::Exponential: RandomizeTableExp(); break;
205 case eDistribution::Midpoint: RandomizeTableMidpoint(); break;
206 } //switch
207} //RandomizeTable

235inline const float CPerlinNoise2D::spline(float x) const{
236 assert(-1.0f <= x && x <= 1.0f);
237
238 float fResult = 0.0f;
239

241 case eSpline::None: fResult = x; break;

243 case eSpline::Quintic: fResult = spline5(x); break;
244 } //switch

299inline const size_t CPerlinNoise2D::hash2(size_t x, size_t y) const{
300 const uint64_t p0 = 43214161; //prime
301 const uint64_t p1 = 43216891; //prime
302 const uint64_t p2 = 73202201; //prime
303
304 const uint64_t h = (p0*(uint64_t)x + p1*(uint64_t)y)%p2; //hash
305
306 return size_t(h >> 8) & m_nMask; //shift and mask
307} //hash2

314void CPerlinNoise2D::HashCorners(size_t x, size_t y, size_t c[4]) const{
315 switch(m_eHash){
316 case eHash::Permutation:
317 c[0] = hash(pair(x, y)); c[1] = hash(pair(x + 1, y));
318 c[2] = hash(pair(x, y + 1)); c[3] = hash(pair(x + 1, y + 1));
319 break;
320
321 case eHash::LinearCongruential:
322 c[0] = hash2(x, y); c[1] = hash2(x + 1, y);
323 c[2] = hash2(x, y + 1); c[3] = hash2(x + 1, y + 1);
324 break;
325
326 case eHash::Std:
327 c[0] = hashstd(pairstd(x, y)); c[1] = hashstd(pairstd(x + 1, y));
328 c[2] = hashstd(pairstd(x, y + 1)); c[3] = hashstd(pairstd(x + 1, y + 1));
329 break;
330 } //switch
331} //HashCorners

344inline const float CPerlinNoise2D::z(size_t h, float x, float y, eNoise t) const{
345 assert(-1.0f <= x && x <= 1.0f);
346 assert(-1.0f <= y && y <= 1.0f);
347 assert(h == (h & m_nMask));
348
349 float result = 0; //return result
350
351 switch(t){ //noise type
352 case eNoise::Perlin:
353 result = x*m_fTable[h] + y*m_fTable[hash(h)]; //gradient times position
354 assert(-2.0f <= result && result <= 2.0f);
355 break;
356
357 case eNoise::Value:
358 result = m_fTable[h]; //get value directly from table
359 assert(-1.0f <= result && result <= 1.0f);
360 break;
361 } //switch
362
363
364 return result;
365} //z

376const float CPerlinNoise2D::Lerp(float sX, float fX, float fY, size_t* c,
377 eNoise t) const
378{
379 assert(-1.0f <= sX && sX <= 1.0f);
380 assert( 0.0f <= fX && fX <= 1.0f);
381 assert(-1.0f <= fY && fY <= 1.0f);
382 assert(-1.0f <= c[0] && c[0] <= 1.0f);
383 assert(-1.0f <= c[1] && c[1] <= 1.0f);
384
385 const float result = lerp(sX, z(c[0], fX, fY, t), z(c[1], fX - 1, fY, t));
386
387 switch(t){ //noise type
388 case eNoise::Perlin:
389 //the worst case in the lerp above is fX and fX - 1 have the same
390 //magnitude, that is, fX == 0.5f, in which case each z gets 1.0f from fY
391 //and 0.5f from fX.
392 assert(-1.5f <= result && result <= 1.5f);
393 return result/1.5f;
394 break;
395
396 case eNoise::Value:
397 assert(-1.0f <= result && result <= 1.0f);
398 return result;
399 break;
400
401 default: return 0.0f;
402 } //switch
403} //Lerp

418const float CPerlinNoise2D::noise(float x, float y, eNoise t) const{
419 const size_t nX = (size_t)floorf(x); //integer part of x
420 const size_t nY = (size_t)floorf(y); //integer part of y
421
422 const float fX = x - floorf(x); //fractional part of x
423 const float fY = y - floorf(y); //fractional part of y
424
425 //smooth fractional parts of x and y using spline curves
426
427 const float sX = spline(fX); //apply spline curve to fractional part of x
428 const float sY = spline(fY); //apply spline curve to fractional part of y
429
430 //hash value at corners of enclosing grid square with integer coordinates
431
432 size_t c[4] = {0}; //for hashed values at corners
433 HashCorners(nX, nY, c); //get hashed values at corners
434
435 //lerp along the top and bottom along the X-axis
436
437 const float a = Lerp(sX, fX, fY, c, t);
438 const float b = Lerp(sX, fX, fY - 1, &(c[2]), t);
439
440 assert(-1.0f <= a && a <= 1.0f);
441 assert(-1.0f <= b && b <= 1.0f);
442
443 //now lerp these values along the Y-axis
444
445 const float result = lerp(sY, a, b);
446 assert(-1.0f <= result && result <= 1.0f);
447 return result;
448} //noise

463const float CPerlinNoise2D::generate(float x, float y, eNoise t, size_t n,
464 float alpha, float beta) const
465{
466 assert(0.0f <= alpha && alpha < 1.0f);
467 assert(beta > 1.0f);
468
469 float sum = 0.0f; //for result
470 float amplitude = 1.0f; //octave amplitude
471
472 for(size_t i=0; i<n; i++){ //for each octave
473 sum += amplitude*noise(x, y, t); //scale noise by amplitude
474 amplitude *= alpha; //reduce amplitude by lacunarity
475 x *= beta; y *= beta; //multiply frequency by persistence
476 } //for
477
478 assert(amplitude == powf(alpha, (float)n));
479
480 float result = (1 - alpha)*sum/(1 - amplitude); //sum of geometric progression
481
482 if(t == eNoise::Perlin)result *= m_fSqrt2; //scale
483
484 assert(-1.0f <= result && result <= 1.0f); //safety
485 return result;
486} //generate
