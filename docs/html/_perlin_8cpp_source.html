80void CPerlinNoise2D::RandomizePermutation(){
81 for(size_t i=0; i<m_nSize; i++) //identity permutation
82 m_nPerm[i] = i;
83
84 for(size_t i=0; i<m_nSize; i++) //randomize
85 std::swap(m_nPerm[i], m_nPerm[i + rand()%(m_nSize - i)]);
86} //RandomizePermutation

106void CPerlinNoise2D::RandomizeTableMidpoint(size_t i, size_t j, float alpha){
107 std::uniform_real_distribution<float> d(-1.0f, 1.0f);
108
109 if(j > i + 1){ //base of recursion is "do nothing"
110 const size_t mid = (i + j)/2; //mid point
111
112 const float fMean = (m_fTable[i] + m_fTable[j - 1])/2.0f; //average of ends
113 const float fRand = d(m_stdRandom); //random value in [-1, 1]
114 m_fTable[mid] = clamp(-1.0f, fMean + fRand, 1.0f); //mid point is average plus offset
115 alpha *= 0.5f; //increase lacunarity
116
117 RandomizeTableMidpoint(i, mid, alpha); //recurse on first half
118 RandomizeTableMidpoint(mid, j, alpha); //recurse on second half
119 } //if
120} //RandomizeTableMidpoint

173void CPerlinNoise2D::RandomizeTableExp(){
174 std::exponential_distribution<float> d(8.0f);
175
176 const size_t half = m_nSize/2;
177
178 for(size_t i=0; i<half; i++) //positive values
179 m_fTable[i] = clamp(0.0f, d(m_stdRandom), 1.0f);
180
181 for(size_t i=half; i<m_nSize; i++) //negative values
182 m_fTable[i] = -clamp(0.0f, d(m_stdRandom), 1.0f);
183} //RandomizeTableExp

190void CPerlinNoise2D::RandomizeTable(eDistribution d){
191 m_stdRandom.seed(timeGetTime()); //reset PRNG
192
193 switch(d){
194 case eDistribution::Uniform: RandomizeTableUniform(); break;
195 case eDistribution::Cosine: RandomizeTableCos(); break;
196 case eDistribution::Normal: RandomizeTableNormal(); break;
197 case eDistribution::Exponential: RandomizeTableExp(); break;
198 case eDistribution::Midpoint: RandomizeTableMidpoint(); break;
199 } //switch
200} //RandomizeTable

251inline const float CPerlinNoise2D::z(size_t h, float x, float y, eNoise t) const{
252 h = h & m_nMask; //safety
253 float result = 0; //return result
254
255 switch(t){ //noise type
256 case eNoise::Perlin:
257 result = x*m_fTable[h] + y*m_fTable[hash(h)]; //gradient times position
258 break;
259
260 case eNoise::Value:
261 result = m_fTable[h]; //get value directly from table
262 break;
263 } //switch
264
265 return clamp(-1.0f, result, 1.0f); //safety
266} //z

316inline const size_t CPerlinNoise2D::hash2(size_t x, size_t y) const{
317 const uint64_t p0 = 43214161; //prime
318 const uint64_t p1 = 43216891; //prime
319 const uint64_t p2 = 73202201; //prime
320
321 const uint64_t h = (p0*(uint64_t)x + p1*(uint64_t)y)%p2; //hash
322
323 return size_t(h >> 8) & m_nMask; //shift and mask
324} //hash2

331void CPerlinNoise2D::HashCorners(size_t x, size_t y, size_t c[4]) const{
332 switch(m_eHash){
333 case eHash::Permutation:
334 c[0] = hash(pair(x, y)); c[1] = hash(pair(x + 1, y));
335 c[2] = hash(pair(x, y + 1)); c[3] = hash(pair(x + 1, y + 1));
336 break;
337
338 case eHash::LinearCongruential:
339 c[0] = hash2(x, y); c[1] = hash2(x + 1, y);
340 c[2] = hash2(x, y + 1); c[3] = hash2(x + 1, y + 1);
341 break;
342
343 case eHash::Std:
344 c[0] = hashstd(pairstd(x, y)); c[1] = hashstd(pairstd(x + 1, y));
345 c[2] = hashstd(pairstd(x, y + 1)); c[3] = hashstd(pairstd(x + 1, y + 1));
346 break;
347 } //switch
348} //HashCorners

378const float CPerlinNoise2D::noise(float x, float y, eNoise t) const{
379 const size_t nX = (size_t)floorf(x); //integer part of x
380 const size_t nY = (size_t)floorf(y); //integer part of y
381
382 const float fX = x - floorf(x); //fractional part of x
383 const float fY = y - floorf(y); //fractional part of y
384
385 //smooth fractional parts of x and y using spline curves
386
387 const float sX = spline(fX); //apply spline curve to fractional part of x
388 const float sY = spline(fY); //apply spline curve to fractional part of y
389
390 //hash value at corners of enclosing grid square with integer coordinates
391
392 size_t c[4] = {0}; //for hashed values at corners
393 HashCorners(nX, nY, c); //get hashed values at corners
394
395 //lerp along the top and bottom along the X-axis
396
397 const float a = Lerp(sX, fX, fY, c, t);
398 const float b = Lerp(sX, fX, fY - 1, &(c[2]), t);
399
400 //now lerp these values along the Y-axis
401
402 return lerp(sY, a, b);
403} //noise

418const float CPerlinNoise2D::generate(float x, float y, eNoise t, size_t n,
419 float alpha, float beta) const
420{
421 float sum = 0.0f; //for result
422 float amplitude = 1.0f;
423
424 for(size_t i=0; i<n; i++){ //for each octave
425 sum += amplitude*noise(x, y, t); //scale noise by amplitude
426 amplitude *= alpha; //reduce amplitude by lacunarity
427 x *= beta; y *= beta; //multiply frequency by persistence
428 } //for
429
430 return (1 - alpha)*sum/(1 - amplitude); //scale by sum of geometric progression
431} //generate
