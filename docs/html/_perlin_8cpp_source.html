80void CPerlinNoise2D::RandomizePermutation(){
81 for(size_t i=0; i<m_nSize; i++) //identity permutation
82 m_nPerm[i] = i;
83
84 for(size_t i=0; i<m_nSize; i++) //randomize
85 std::swap(m_nPerm[i], m_nPerm[i + rand()%(m_nSize - i)]);
86} //RandomizePermutation

106void CPerlinNoise2D::RandomizeTableMidpoint(size_t i, size_t j, float alpha){
107 std::uniform_real_distribution<float> d(-1.0f, 1.0f);
108
109 if(j > i + 1){ //base of recursion is "do nothing"
110 const size_t mid = (i + j)/2; //mid point
111
112 const float fMean = (m_fTable[i] + m_fTable[j - 1])/2.0f; //average of ends
113 const float fRand = alpha*d(m_stdRandom); //random offset
114 m_fTable[mid] = clamp(-1.0f, fMean + fRand, 1.0f); //mid point is average plus offset
115 alpha *= 0.5f; //increase lacunarity
116
117 RandomizeTableMidpoint(i, mid, alpha); //recurse on first half
118 RandomizeTableMidpoint(mid, j, alpha); //recurse on second half
119 } //if
120} //RandomizeTableMidpoint

193void CPerlinNoise2D::RandomizeTableExp(){
194 std::exponential_distribution<float> d(8.0f);
195
196 const size_t half = m_nSize/2;
197
198 for(size_t i=0; i<half; i++) //positive values
199 m_fTable[i] = clamp(0.0f, d(m_stdRandom), 1.0f);
200
201 for(size_t i=half; i<m_nSize; i++) //negative values
202 m_fTable[i] = -clamp(0.0f, d(m_stdRandom), 1.0f);
203} //RandomizeTableExp

210void CPerlinNoise2D::RandomizeTable(eDistribution d){
211 m_stdRandom.seed(timeGetTime()); //reset PRNG
212
213 switch(d){
214 case eDistribution::Uniform: RandomizeTableUniform(); break;
215 case eDistribution::Maximal: RandomizeTableMaximal(); break;
216 case eDistribution::Cosine: RandomizeTableCos(); break;
217 case eDistribution::Normal: RandomizeTableNormal(); break;
218 case eDistribution::Exponential: RandomizeTableExp(); break;
219 case eDistribution::Midpoint: RandomizeTableMidpoint(); break;
220 } //switch
221} //RandomizeTable

249inline const float CPerlinNoise2D::spline(float x) const{
250 assert(-1.0f <= x && x <= 1.0f);
251
252 float fResult = 0.0f;
253

255 case eSpline::None: fResult = x; break;

257 case eSpline::Quintic: fResult = spline5(x); break;

313inline const size_t CPerlinNoise2D::hash2(size_t x, size_t y) const{
314 const uint64_t p0 = 43214161; //prime
315 const uint64_t p1 = 43216891; //prime
316 const uint64_t p2 = 73202201; //prime
317
318 const uint64_t h = (p0*(uint64_t)x + p1*(uint64_t)y)%p2; //hash
319
320 return size_t(h >> 8) & m_nMask; //shift and mask
321} //hash2

328void CPerlinNoise2D::HashCorners(size_t x, size_t y, size_t c[4]) const{
329 switch(m_eHash){
330 case eHash::Permutation:
331 c[0] = hash(pair(x, y)); c[1] = hash(pair(x + 1, y));
332 c[2] = hash(pair(x, y + 1)); c[3] = hash(pair(x + 1, y + 1));
333 break;
334
335 case eHash::LinearCongruential:
336 c[0] = hash2(x, y); c[1] = hash2(x + 1, y);
337 c[2] = hash2(x, y + 1); c[3] = hash2(x + 1, y + 1);
338 break;
339
340 case eHash::Std:
341 c[0] = hashstd(pairstd(x, y)); c[1] = hashstd(pairstd(x + 1, y));
342 c[2] = hashstd(pairstd(x, y + 1)); c[3] = hashstd(pairstd(x + 1, y + 1));
343 break;
344 } //switch
345} //HashCorners

358inline const float CPerlinNoise2D::z(size_t h, float x, float y, eNoise t) const{
359 assert(-1.0f <= x && x <= 1.0f);
360 assert(-1.0f <= y && y <= 1.0f);
361 assert(h == (h & m_nMask));
362
363 float result = 0; //return result
364
365 switch(t){ //noise type
366 case eNoise::Perlin:
367 result = x*m_fTable[h] + y*m_fTable[hash(h)]; //gradient times position
368 assert(-2.0f <= result && result <= 2.0f);
369 break;
370
371 case eNoise::Value:
372 result = m_fTable[h]; //get value directly from table
373 assert(-1.0f <= result && result <= 1.0f);
374 break;
375 } //switch
376
377 return result;
378} //z

389const float CPerlinNoise2D::Lerp(float sX, float fX, float fY, size_t* c,
390 eNoise t) const
391{
392 assert(-1.0f <= sX && sX <= 1.0f);
393 assert( 0.0f <= fX && fX <= 1.0f);
394 assert(-1.0f <= fY && fY <= 1.0f);
395
396 const float result = lerp(sX, z(c[0], fX, fY, t), z(c[1], fX - 1, fY, t));
397
398 switch(t){ //noise type
399 case eNoise::Perlin:
400 //the worst case in the lerp above is fX and fX - 1 have the same
401 //magnitude, that is, fX == 0.5f, in which case each z gets 1.0f from fY
402 //and 0.5f from fX.
403 assert(-1.5f <= result && result <= 1.5f);
404 //return result/1.5f;
405 return result;
406 break;
407
408 case eNoise::Value:
409 assert(-1.0f <= result && result <= 1.0f);
410 return result;
411 break;
412
413 default: return 0.0f;
414 } //switch
415} //Lerp

430const float CPerlinNoise2D::noise(float x, float y, eNoise t) const{
431 const size_t nX = (size_t)floorf(x); //integer part of x
432 const size_t nY = (size_t)floorf(y); //integer part of y
433
434 const float fX = x - floorf(x); //fractional part of x
435 const float fY = y - floorf(y); //fractional part of y
436
437 //smooth fractional parts of x and y using spline curves
438
439 const float sX = spline(fX); //apply spline curve to fractional part of x
440 const float sY = spline(fY); //apply spline curve to fractional part of y
441
442 //hash value at corners of enclosing grid square with integer coordinates
443
444 size_t c[4] = {0}; //for hashed values at corners
445 HashCorners(nX, nY, c); //get hashed values at corners
446
447 //lerp along the top and bottom along the X-axis
448
449 const float a = Lerp(sX, fX, fY, c, t);
450 const float b = Lerp(sX, fX, fY - 1, &(c[2]), t);
451
452 //now lerp these values along the Y-axis
453
454 const float result = lerp(sY, a, b);
455 assert(-1.0f <= result && result <= 1.0f);
456 return result;
457} //noise

472const float CPerlinNoise2D::generate(float x, float y, eNoise t, size_t n,
473 float alpha, float beta) const
474{
475 assert(0.0f <= alpha && alpha < 1.0f);
476 assert(beta > 1.0f);
477
478 float sum = 0.0f; //for result
479 float amplitude = 1.0f; //octave amplitude
480
481 for(size_t i=0; i<n; i++){ //for each octave
482 sum += amplitude*noise(x, y, t); //scale noise by amplitude
483 amplitude *= alpha; //reduce amplitude by lacunarity
484 x *= beta; y *= beta; //multiply frequency by persistence
485 } //for
486
487 assert(amplitude == powf(alpha, (float)n));
488
489 float result = (1 - alpha)*sum/(1 - amplitude); //sum of geometric progression
490
491 if(t == eNoise::Perlin)result *= m_fSqrt2; //scale
492
493 assert(-1.0f <= result && result <= 1.0f); //safety
494 return result;
495} //generate
