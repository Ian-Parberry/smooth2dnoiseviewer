65void CMain::OnPaint(){
66 PAINTSTRUCT ps; //paint structure
67 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
68 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
69
70 //bitmap width and height
71
72 const int nBitmapWidth = m_pBitmap->GetWidth();
73 const int nBitmapHeight = m_pBitmap->GetHeight();
74
75 //get client rectangle
76
77 RECT rectClient; //for client rectangle
78 GetClientRect(m_hWnd, &rectClient); //get client rectangle
79 const int nClientWidth = rectClient.right - rectClient.left; //client width
80 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
81
82 //compute destination rectangle
83
84 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
85
86 const int width = min(nDestSide, nBitmapWidth); //dest rect width
87 const int height = min(nDestSide, nBitmapHeight); //dest rect height
88
89 const int x = max(0, nClientWidth - width)/2; //x margin
90 const int y = max(0, nClientHeight - height)/2; //y margin
91
92 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
93
94 //draw image to destination rectangle then clean up
95
96 graphics.DrawImage(m_pBitmap, rectDest);
97
98 EndPaint(m_hWnd, &ps); //this must be done last
99} //OnPaint

165void CMain::GeneratePerlinNoise(){
166 initPerlin();
167 const float fScale = 64.0f;
168
169 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
170 for(UINT j=0; j<m_pBitmap->GetHeight(); j++){
171 const float fPerlin = PerlinNoise2D(i/fScale, j/fScale, 0.5f, 2.0f, 4);
172 const BYTE gray = BYTE(float(0xFF)*(fPerlin/2.0f + 0.5f));
173 m_pBitmap->SetPixel(i, j, Gdiplus::Color(255, gray, gray, gray));
174 } //for
175} //GeneratePerlinNoise
