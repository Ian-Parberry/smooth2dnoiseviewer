67void CMain::OnPaint(){
68 PAINTSTRUCT ps; //paint structure
69 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
70 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
71
72 //get bitmap width and height
73
74 const int nBitmapWidth = m_pBitmap->GetWidth();
75 const int nBitmapHeight = m_pBitmap->GetHeight();
76
77 //get client rectangle
78
79 RECT rectClient; //for client rectangle
80 GetClientRect(m_hWnd, &rectClient); //get client rectangle
81 const int nClientWidth = rectClient.right - rectClient.left; //client width
82 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
83
84 //compute destination rectangle
85
86 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
87
88 const int width = min(nDestSide, nBitmapWidth); //dest rect width
89 const int height = min(nDestSide, nBitmapHeight); //dest rect height
90
91 const int x = max(0, nClientWidth - width)/2; //x margin
92 const int y = max(0, nClientHeight - height)/2; //y margin
93
94 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
95
96 //draw image to destination rectangle then clean up
97
98 graphics.DrawImage(m_pBitmap, rectDest);
99
100 EndPaint(m_hWnd, &ps); //this must be done last
101} //OnPaint

230void CMain::GenerateNoiseBitmap(eNoise t){
231 m_eNoise = t; //remember the noise type
232 UpdateMenus(); //changing noise type may change the menu status
233
234 m_fMin = 1000.0f; //init noise minima to something stupidly large
235 m_fMax = -1000.0f; //init noise maxima to something stupidly small
236 m_fAve = 0.0f; //init sum for average
237
238 const UINT w = m_pBitmap->GetWidth(); //bitmap width
239 const UINT h = m_pBitmap->GetHeight(); //bitmap height
240
241 for(UINT i=0; i<w; i++){
242 const float x = m_fOriginX + i/m_fScale; //noise X-coordinate
243
244 for(UINT j=0; j<h; j++){
245 const float y = m_fOriginY + j/m_fScale; //noise Y-coordinate
246 const float noise = m_pPerlin->generate(x, y, t, m_nOctaves); //noise
247 SetPixel(i, j, noise); //draw noise pixel to bitmap
248
249 //recompute maximum, minimum, and average
250 m_fMin = min(m_fMin, noise);
251 m_fMax = max(m_fMax, noise);
252 m_fAve += noise;
253 } //for
254 } //for
255
256 m_fAve /= w*h; //compute average from sum
257
258 if(m_bShowGrid)DrawGrid();
259 if(m_bShowCoords)DrawCoords();
260} //GenerateNoiseBitmap

270void CMain::GenerateNoiseBitmap(Gdiplus::PointF point, Gdiplus::RectF rect){
271 const UINT nLeft = (UINT)floorf(rect.GetLeft() + point.X);
272 const UINT nRight = (UINT)ceilf(rect.GetRight() + point.X);
273 const UINT nTop = (UINT)floorf(rect.GetTop() + point.Y);
274 const UINT nBottom = (UINT)ceilf(rect.GetBottom() + point.Y);
275
276 for(UINT i=nLeft; i<nRight; i++){
277 const float x = m_fOriginX + i/m_fScale;
278
279 for(UINT j=nTop; j<nBottom; j++){
280 const float y = m_fOriginY + j/m_fScale;
281 SetPixel(i, j, m_pPerlin->generate(x, y, m_eNoise, m_nOctaves));
282 } //for
283 } //for
284} //GenerateNoiseBitmap

294void CMain::DrawCoords(){
295 Gdiplus::FontFamily ff(L"Arial");
296 Gdiplus::Font font(&ff, 20, Gdiplus::FontStyleRegular, Gdiplus::UnitPixel);
297 Gdiplus::SolidBrush brush(Gdiplus::Color::White);
298
299 Gdiplus::Graphics graphics(m_pBitmap);
300
301 //top left corner coordinates
302
303 Gdiplus::PointF point(0.0f, 0.0f); //text position on screen
304
305 std::wstring wstr = L"("; //text of origin coordinates
306 wstr += std::to_wstring((size_t)floorf(m_fOriginX)) + L", ";
307 wstr += std::to_wstring((size_t)floorf(m_fOriginY)) + L")";
308
309 Gdiplus::RectF rect, unused;
310 graphics.MeasureString(wstr.c_str(), -1, &font, unused, &rect); //measure text
311
312 if(m_bShowCoords) //draw
313 graphics.DrawString(wstr.c_str(), -1, &font, point, &brush);
314 else GenerateNoiseBitmap(point, rect); //undraw
315
316 //bottom right corner coordinates
317
318 const float x = m_fOriginX + m_pBitmap->GetWidth()/m_fScale; //x-coordinate
319 const float y = m_fOriginY + m_pBitmap->GetHeight()/m_fScale; //y-coordinate
320
321 wstr = L"(" + to_wstring_f(x, 2) + L", " + to_wstring_f(y, 2) + L")"; //text
322
323 graphics.MeasureString(wstr.c_str(), -1, &font, unused, &rect); //measure text
324 point.X = m_pBitmap->GetWidth() - rect.Width; //text x-coordinate on screen
325 point.Y = m_pBitmap->GetHeight() - rect.Height; //text y-coordinate on screen
326
327 if(m_bShowCoords) //draw
328 graphics.DrawString(wstr.c_str(), -1, &font, point, &brush);
329 else GenerateNoiseBitmap(point, rect); //undraw
330
331 if(m_bShowGrid && !m_bShowCoords)
332 DrawGrid(); //in case we erased part of the grid lines
333} //DrawCoords

341void CMain::DrawGrid(){
342 const float fBitmapWidth = (float)m_pBitmap->GetWidth();
343 const float fBitmapHeight = (float)m_pBitmap->GetHeight();
344
345 const UINT nv = (UINT)floorf(fBitmapWidth/m_fScale); //number of vertical lines
346 const UINT nh = (UINT)floorf(fBitmapHeight/m_fScale); //number of horizontal lines
347
348 Gdiplus::Graphics graphics(m_pBitmap);
349 Gdiplus::Pen pen(Gdiplus::Color(255, 0, 255, 0)); //green
350
351 //horizontal lines
352
353 Gdiplus::PointF left(0.0f, m_fScale); //left X-coordinate of line
354 Gdiplus::PointF right(fBitmapWidth, m_fScale); //right X-coordinate of line
355 Gdiplus::RectF rect(0.0f, 0.0f, fBitmapWidth, 1.0f); //bounding rectangle
356
357 for(UINT i=0; i<nh; i++){ //for each horizontal line
358 if(m_bShowGrid)graphics.DrawLine(&pen, left, right); //draw line
359 else GenerateNoiseBitmap(left, rect); //undraw line
360
361 //move down to next horizontal line
362 left.Y += m_fScale;
363 right.Y += m_fScale;
364 } //for
365
366 //vertical lines
367
368 Gdiplus::PointF top(m_fScale, 0.0f); //top Y-coordinate of line
369 Gdiplus::PointF bottom(m_fScale, fBitmapHeight); //bottom Y-coordinate of line
370 rect = Gdiplus::RectF(0.0f, 0.0f, 1.0f, fBitmapHeight); //bounding rectangle
371
372 for(UINT i=0; i<nv; i++){ //for each vertical line
373 if(m_bShowGrid)graphics.DrawLine(&pen, top, bottom); //draw line
374 else GenerateNoiseBitmap(top, rect); //undraw line
375
376 //move right to next vertical line
377 top.X += m_fScale;
378 bottom.X += m_fScale;
379 } //for
380
381 if(!m_bShowGrid && m_bShowCoords)
382 DrawCoords(); //in case we erased some of the coordinate text
383} //DrawGrid

556const std::wstring CMain::GetFileName() const{
557 std::wstring wstr;
558
559 switch(m_eNoise){
560 case eNoise::Perlin: wstr = L"Perlin"; break;
561 case eNoise::Value: wstr = L"Value"; break;
562 } //switch
563
564 switch(m_pPerlin->GetHash()){
565 case eHash::Permutation: wstr += L"-Perm"; break;
566 case eHash::LinearCongruential: wstr += L"-Lin"; break;
567 case eHash::Std: wstr += L"-Std"; break;
568 } //switch
569
570 switch(m_pPerlin->GetDistribution()){
571 case eDistribution::Uniform: break; //nothing, which is the default
572 case eDistribution::Maximal: wstr += L"-Max"; break;
573 case eDistribution::Cosine: wstr += L"-Cos"; break;
574 case eDistribution::Normal: wstr += L"-Norm"; break;
575 case eDistribution::Exponential: wstr += L"-Exp"; break;
576 case eDistribution::Midpoint: wstr += L"-Mid"; break;
577 } //switch
578
579 switch(m_pPerlin->GetSpline()){
580 case eSpline::None: wstr += L"-NoSpline"; break;
581 case eSpline::Cubic: break; //nothing, which is the default
582 case eSpline::Quintic: wstr += L"-Quintic"; break;
583 } //switch
584
585 wstr += L"-" + std::to_wstring(m_nOctaves);
586 wstr += L"-" + std::to_wstring(m_pPerlin->GetTableSize());
587 wstr += L"-" + std::to_wstring((size_t)round(m_fScale));
588
589 return wstr;
590} //GetFileName

595const std::wstring CMain::GetNoiseDescription() const{
596 std::wstring wstr; //for noise description
597
598 //number of octaves
599
600 if(m_eNoise == eNoise::Perlin || m_eNoise == eNoise::Value){
601 wstr += std::to_wstring(m_nOctaves) + L" octave";
602 if(m_nOctaves > 1)wstr += L"s";
603 wstr += L" of ";
604 } //if
605
606 //type of noise
607
608 switch(m_eNoise){
609 case eNoise::Perlin: wstr += L"Perlin"; break;
610 case eNoise::Value: wstr += L"Value"; break;
611 } //switch
612
613 wstr += L" Noise";
614
615 //origin
616
617 wstr += L" with origin (";
618 wstr += to_wstring_f(m_fOriginX, 2) + L", ";
619 wstr += to_wstring_f(m_fOriginY, 2) + L"), ";
620
621 //hash function
622
623 switch(m_pPerlin->GetHash()){
624 case eHash::Permutation: wstr += L"a permutation"; break;
625 case eHash::LinearCongruential: wstr += L"linear congruential"; break;
626 case eHash::Std: wstr += L"std"; break;
627 } //switch
628
629 wstr += L" hash function, ";
630
631 //distribution
632
633 switch(m_pPerlin->GetDistribution()){
634 case eDistribution::Uniform: wstr += L"uniform"; break;
635 case eDistribution::Maximal: wstr += L"maximal"; break;
636 case eDistribution::Cosine: wstr += L"cosine"; break;
637 case eDistribution::Normal: wstr += L"normal"; break;
638 case eDistribution::Exponential: wstr += L"exponential"; break;
639 case eDistribution::Midpoint: wstr += L"midpoint displacement"; break;
640 } //switch
641
642 switch(m_eNoise){
643 case eNoise::Perlin: wstr += L" gradient"; break;
644 case eNoise::Value: wstr += L" height"; break;
645 } //switch
646
647 wstr += L" distribution, ";
648
649 //spline function
650
651 switch(m_pPerlin->GetSpline()){
652 case eSpline::None: wstr += L"no"; break;
653 case eSpline::Cubic: wstr += L"cubic"; break;
654 case eSpline::Quintic: wstr += L"quintic"; break;
655 } //switch
656
657 wstr += L" spline function, ";
658
659 //scale
660
661 wstr += L"scale " + std::to_wstring((size_t)round(m_fScale)) + L", and ";
662
663 //table size
664
665 if(m_pPerlin->GetHash() == eHash::Permutation)
666 wstr += L"permutation and ";
667
668 switch(m_eNoise){
669 case eNoise::Perlin: wstr += L"gradient "; break;
670 case eNoise::Value: wstr += L"value "; break;
671 } //switch
672
673 wstr += L"table size ";
674 wstr += std::to_wstring(m_pPerlin->GetTableSize());
675 wstr += L". ";
676
677 //noise max, min, and average
678
679 wstr += L"Largest generated noise ";
680 wstr += to_wstring_f(m_fMax, 4) + L". ";
681 wstr += L"Smallest generated noise ";
682 wstr += to_wstring_f(m_fMin, 4) + L". ";
683 wstr += L"Average generated noise ";
684 wstr += to_wstring_f(m_fAve, 4) + L".";
685
686 return wstr;
687} //GetNoiseDescription
