69void CMain::OnPaint(){
70 PAINTSTRUCT ps; //paint structure
71 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
72 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
73
74 //bitmap width and height
75
76 const int nBitmapWidth = m_pBitmap->GetWidth();
77 const int nBitmapHeight = m_pBitmap->GetHeight();
78
79 //get client rectangle
80
81 RECT rectClient; //for client rectangle
82 GetClientRect(m_hWnd, &rectClient); //get client rectangle
83 const int nClientWidth = rectClient.right - rectClient.left; //client width
84 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
85
86 //compute destination rectangle
87
88 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
89
90 const int width = min(nDestSide, nBitmapWidth); //dest rect width
91 const int height = min(nDestSide, nBitmapHeight); //dest rect height
92
93 const int x = max(0, nClientWidth - width)/2; //x margin
94 const int y = max(0, nClientHeight - height)/2; //y margin
95
96 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
97
98 //draw image to destination rectangle then clean up
99
100 graphics.DrawImage(m_pBitmap, rectDest);
101
102 EndPaint(m_hWnd, &ps); //this must be done last
103} //OnPaint

194void CMain::GeneratePerlinNoise(eNoise t){
195 const float fScale = 64.0f;
196
197 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
198 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
199 SetPixel(i, j, m_pPerlin->generate(i/fScale, j/fScale, t, 0.5f, 2.0f, 4));
200} //GeneratePerlinNoise
