71void CMain::OnPaint(){
72 PAINTSTRUCT ps; //paint structure
73 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
74 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
75
76 //bitmap width and height
77
78 const int nBitmapWidth = m_pBitmap->GetWidth();
79 const int nBitmapHeight = m_pBitmap->GetHeight();
80
81 //get client rectangle
82
83 RECT rectClient; //for client rectangle
84 GetClientRect(m_hWnd, &rectClient); //get client rectangle
85 const int nClientWidth = rectClient.right - rectClient.left; //client width
86 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
87
88 //compute destination rectangle
89
90 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
91
92 const int width = min(nDestSide, nBitmapWidth); //dest rect width
93 const int height = min(nDestSide, nBitmapHeight); //dest rect height
94
95 const int x = max(0, nClientWidth - width)/2; //x margin
96 const int y = max(0, nClientHeight - height)/2; //y margin
97
98 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
99
100 //draw image to destination rectangle then clean up
101
102 graphics.DrawImage(m_pBitmap, rectDest);
103
104 EndPaint(m_hWnd, &ps); //this must be done last
105} //OnPaint

201void CMain::GeneratePixelNoise(){
202 m_eNoise = eNoise::Pixel;
203
204 EnableMenuItem(m_hFileMenu, IDM_FILE_SAVE, MF_ENABLED);
205 CheckMenuItem(m_hGenMenu, IDM_GENERATE_PIXELNOISE, MF_CHECKED);
206 CheckMenuItem(m_hGenMenu, IDM_GENERATE_PERLINNOISE, MF_UNCHECKED);
207 CheckMenuItem(m_hGenMenu, IDM_GENERATE_VALUENOISE, MF_UNCHECKED);
208 EnableMenuItem(m_hGenMenu, IDM_GENERATE_RANDOMIZE, MF_ENABLED);
209 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_CUBIC, MF_GRAYED);
210 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_QUINTIC, MF_GRAYED);
211
212 std::default_random_engine g;
213 g.seed(m_nSeed);
214
215 switch(m_eCurDist){
216 case eDistribution::Uniform: {
217 std::uniform_real_distribution<float> d(-1.0f, 1.0f);
218
219 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
220 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
221 SetPixel(i, j, d(g));
222 } //case
223 break;
224
225 case eDistribution::Cosine: {
226 std::uniform_real_distribution<float> d(0.0f, PI);
227
228 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
229 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
230 SetPixel(i, j, cosf(d(g)));
231 } //case
232 break;
233
234 case eDistribution::Normal:{
235 std::normal_distribution<float> d(500.0f, 200.0f);
236
237 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
238 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
239 SetPixel(i, j, 2.0f*clamp(0.0f, d(g)/1000.0f, 1.0f) - 1.0f);
240 } //case
241 break;
242
243 case eDistribution::Exponential: {
244 std::exponential_distribution<float> d(3.5f);
245
246 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
247 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
248 SetPixel(i, j, ((rand() & 1)? -1.0f: 1.0f)*clamp(0.0f, d(g), 1.0f));
249 } //case
250 break;
251 } //switch
252} //GeneratePixelNoise

259void CMain::GeneratePerlinNoise(eNoise t){
260 m_eNoise = t;
261
262 EnableMenuItem(m_hFileMenu, IDM_FILE_SAVE, MF_ENABLED);
263 CheckMenuItem(m_hGenMenu, IDM_GENERATE_PIXELNOISE, MF_UNCHECKED);
264 CheckMenuItem(m_hGenMenu, IDM_GENERATE_PERLINNOISE,
265 (t == eNoise::Perlin)? MF_CHECKED: MF_UNCHECKED);
266 CheckMenuItem(m_hGenMenu, IDM_GENERATE_VALUENOISE,
267 (t == eNoise::Value)? MF_CHECKED: MF_UNCHECKED);
268 EnableMenuItem(m_hGenMenu, IDM_GENERATE_RANDOMIZE, MF_ENABLED);
269 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_CUBIC, MF_ENABLED);
270 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_QUINTIC, MF_ENABLED);
271
272 const float fScale = 64.0f;
273
274 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
275 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
276 SetPixel(i, j, m_pPerlin->generate(i/fScale, j/fScale, t, 0.5f, 2.0f, 4));
277} //GeneratePerlinNoise

304void CMain::Regenerate(){
305 switch(m_eNoise){
306 case eNoise::Pixel:
307 GeneratePixelNoise();
308 break;
309
310 case eNoise::Perlin:
311 case eNoise::Value:
312 GeneratePerlinNoise(m_eNoise);
313 break;
314
315 default:
316 Clear();
317 break;
318 } //switch
319} //Regenerate

379const std::wstring CMain::GetFileName() const{
380 std::wstring wstr;
381
382 switch(m_eNoise){
383 case eNoise::Pixel: wstr = L"Pixel"; break;
384 case eNoise::Perlin: wstr = L"Perlin"; break;
385 case eNoise::Value: wstr = L"Value"; break;
386 } //switch
387
388 switch(m_eCurDist){
389 case eDistribution::Uniform: break; //nothing
390 case eDistribution::Cosine: wstr += L"-Cos"; break;
391 case eDistribution::Normal: wstr += L"-Norm"; break;
392 case eDistribution::Exponential: wstr += L"-Exp"; break;
393 } //switch
394
395 if(m_eNoise != eNoise::Pixel)
396 switch(m_eCurSpline){
397 case eSpline::Cubic: break; //nothing
398 case eSpline::Quintic:wstr += L"-Quintic"; break;
399 } //switch
400
401 return wstr;
402} //GetFileName
