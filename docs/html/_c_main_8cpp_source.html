68void CMain::OnPaint(){
69 PAINTSTRUCT ps; //paint structure
70 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
71 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
72
73 //get bitmap width and height
74
75 const int nBitmapWidth = m_pBitmap->GetWidth();
76 const int nBitmapHeight = m_pBitmap->GetHeight();
77
78 //get client rectangle
79
80 RECT rectClient; //for client rectangle
81 GetClientRect(m_hWnd, &rectClient); //get client rectangle
82 const int nClientWidth = rectClient.right - rectClient.left; //client width
83 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
84
85 //compute destination rectangle
86
87 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
88
89 const int width = min(nDestSide, nBitmapWidth); //dest rect width
90 const int height = min(nDestSide, nBitmapHeight); //dest rect height
91
92 const int x = max(0, nClientWidth - width)/2; //x margin
93 const int y = max(0, nClientHeight - height)/2; //y margin
94
95 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
96
97 //draw image to destination rectangle then clean up
98
99 graphics.DrawImage(m_pBitmap, rectDest);
100
101 EndPaint(m_hWnd, &ps); //this must be done last
102} //OnPaint

230void CMain::GenerateNoiseBitmap(eNoise t){
231 m_eNoise = t; //remember the noise type
232 UpdateMenus(); //changing noise type may change the menu status
233
234 for(UINT i=0; i<m_pBitmap->GetWidth(); i++){
235 const float x = m_fOriginX + i/m_fScale;
236
237 for(UINT j=0; j<m_pBitmap->GetHeight(); j++){
238 const float y = m_fOriginY + j/m_fScale;
239 const float z = m_pPerlin->generate(x, y, t, m_nOctaves);
240 SetPixel(i, j, z);
241 } //for
242 } //for
243
244 if(m_bShowGrid)DrawGrid();
245 if(m_bShowCoords)DrawCoords();
246} //GenerateNoiseBitmap

252void CMain::DrawCoords(){
253 Gdiplus::FontFamily ff(L"Arial");
254 Gdiplus::Font font(&ff, 20, Gdiplus::FontStyleRegular, Gdiplus::UnitPixel);
255 Gdiplus::SolidBrush brush(Gdiplus::Color::White);
256
257 Gdiplus::Graphics graphics(m_pBitmap);
258
259 //top left corner
260
261 Gdiplus::PointF pointF(0.0f, 0.0f); //text position on screen
262
263 std::wstring wstr = L"("; //text of origin coordinates
264 wstr += std::to_wstring((size_t)floorf(m_fOriginX)) + L", ";
265 wstr += std::to_wstring((size_t)floorf(m_fOriginY)) + L")";
266
267 graphics.DrawString(wstr.c_str(), -1, &font, pointF, &brush);
268
269 //bottom right corner
270
271 const float x = m_fOriginX + m_pBitmap->GetWidth()/m_fScale; //x-coordinate
272 const float y = m_fOriginY + m_pBitmap->GetHeight()/m_fScale; //y-coordinate
273
274 wstr = L"(" + to_wstring_f(x, 2) + L", " + to_wstring_f(y, 2) + L")"; //text
275
276 Gdiplus::RectF rec, unused;
277 graphics.MeasureString(wstr.c_str(), -1, &font, unused, &rec); //measure text
278 const float w = ceilf(rec.Width); //width of text on screen
279 const float h = ceilf(rec.Height); //height of text on screen
280
281 pointF.X = m_pBitmap->GetWidth() - w; //text x-coordinate on screen
282 pointF.Y = m_pBitmap->GetHeight() - h; //text y-coordinate on screen
283
284 graphics.DrawString(wstr.c_str(), -1, &font, pointF, &brush);
285} //DrawCoords

289void CMain::DrawGrid(){
290 const float fBitmapWidth = (float)m_pBitmap->GetWidth();
291 const float fBitmapHeight = (float)m_pBitmap->GetHeight();
292
293 const UINT nv = (UINT)floorf(fBitmapWidth/m_fScale); //number of vertical lines
294 const UINT nh = (UINT)floorf(fBitmapHeight/m_fScale); //number of horizontal lines
295
296 Gdiplus::Graphics graphics(m_pBitmap);
297 Gdiplus::Pen pen(Gdiplus::Color(255, 0, 255, 0)); //green
298
299 //horizontal lines
300
301 Gdiplus::PointF left(0.0f, m_fScale);
302 Gdiplus::PointF right(fBitmapWidth, m_fScale);
303
304 for(UINT i=0; i<nh; i++){
305 graphics.DrawLine(&pen, left, right);
306 left.Y += m_fScale;
307 right.Y += m_fScale;
308 } //for
309
310 //vertical lines
311
312 Gdiplus::PointF top(m_fScale, 0.0f);
313 Gdiplus::PointF bottom(m_fScale, fBitmapHeight);
314
315 for(UINT i=0; i<nv; i++){
316 graphics.DrawLine(&pen, top, bottom);
317 top.X += m_fScale;
318 bottom.X += m_fScale;
319 } //for
320} //DrawGrid

484const std::wstring CMain::GetFileName() const{
485 std::wstring wstr;
486
487 switch(m_eNoise){
488 case eNoise::Perlin: wstr = L"Perlin"; break;
489 case eNoise::Value: wstr = L"Value"; break;
490 } //switch
491
492 switch(m_eHash){
493 case eHash::Permutation: wstr += L"-Perm"; break;
494 case eHash::LinearCongruential: wstr += L"-Lin"; break;
495 case eHash::Std: wstr += L"-Std"; break;
496 } //switch
497
498 switch(m_eDistr){
499 case eDistribution::Uniform: break; //nothing, which is the default
500 case eDistribution::Cosine: wstr += L"-Cos"; break;
501 case eDistribution::Normal: wstr += L"-Norm"; break;
502 case eDistribution::Exponential: wstr += L"-Exp"; break;
503 case eDistribution::Midpoint: wstr += L"-Mid"; break;
504 } //switch
505
506 switch(m_eSpline){
507 case eSpline::None: wstr += L"-NoSpline"; break;
508 case eSpline::Cubic: break; //nothing, which is the default
509 case eSpline::Quintic: wstr += L"-Quintic"; break;
510 } //switch
511
512 wstr += L"-" + std::to_wstring(m_nOctaves);
513 wstr += L"-" + std::to_wstring(1 << m_nLog2TableSize);
514 wstr += L"-" + std::to_wstring((size_t)round(m_fScale));
515
516 return wstr;
517} //GetFileName

522const std::wstring CMain::GetNoiseDescription() const{
523 std::wstring wstr; //for noise description
524
525 //number of octaves
526
527 if(m_eNoise == eNoise::Perlin || m_eNoise == eNoise::Value){
528 wstr += std::to_wstring(m_nOctaves) + L" octave";
529 if(m_nOctaves > 1)wstr += L"s";
530 wstr += L" of ";
531 } //if
532
533 //type of noise
534
535 switch(m_eNoise){
536 case eNoise::Perlin: wstr += L"Perlin"; break;
537 case eNoise::Value: wstr += L"Value"; break;
538 } //switch
539
540 wstr += L" Noise";
541
542 //origin
543
544 wstr += L" with origin (";
545 wstr += to_wstring_f(m_fOriginX, 2) + L", ";
546 wstr += to_wstring_f(m_fOriginY, 2) + L"), ";
547
548 //hash function
549
550 switch(m_eHash){
551 case eHash::Permutation: wstr += L"a permutation"; break;
552 case eHash::LinearCongruential: wstr += L"linear congruential"; break;
553 case eHash::Std: wstr += L"std"; break;
554 } //switch
555
556 wstr += L" hash function, ";
557
558 //distribution
559
560 switch(m_eDistr){
561 case eDistribution::Uniform: wstr += L"uniform"; break;
562 case eDistribution::Cosine: wstr += L"cosine"; break;
563 case eDistribution::Normal: wstr += L"normal"; break;
564 case eDistribution::Exponential: wstr += L"exponential"; break;
565 case eDistribution::Midpoint: wstr += L"midpoint displacement"; break;
566 } //switch
567
568 switch(m_eNoise){
569 case eNoise::Perlin: wstr += L" gradient"; break;
570 case eNoise::Value: wstr += L" height"; break;
571 } //switch
572
573 wstr += L" distribution, ";
574
575 //spline function
576
577 switch(m_eSpline){
578 case eSpline::None: wstr += L"no"; break;
579 case eSpline::Cubic: wstr += L"cubic"; break;
580 case eSpline::Quintic: wstr += L"quintic"; break;
581 } //switch
582
583 wstr += L" spline function, ";
584
585 //scale
586
587 wstr += L"scale " + std::to_wstring((size_t)round(m_fScale)) + L", and ";
588
589 //table size
590
591 if(m_eHash == eHash::Permutation)
592 wstr += L"permutation and ";
593
594 switch(m_eNoise){
595 case eNoise::Perlin: wstr += L"gradient "; break;
596 case eNoise::Value: wstr += L"value "; break;
597 } //switch
598
599 wstr += L"table size ";
600 wstr += std::to_wstring(1 << m_nLog2TableSize);
601
602 //lacunarity and persistence
603
604 wstr += L". Lacunarity and persistence are fixed at 0.5 and 2.0, respectively.";
605
606 return wstr;
607} //GetNoiseDescription
