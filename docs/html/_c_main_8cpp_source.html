230 void CMain::GenerateNoiseBitmap(eNoise t){
231  m_eNoise = t; //remember the noise type
232  UpdateMenus(); //changing noise type may change the menu status
233 
234  m_fMin = 1000.0f; //init noise minima to something stupidly large
235  m_fMax = -1000.0f; //init noise maxima to something stupidly small
236  m_fAve = 0.0f; //init sum for average
237 
238  const UINT w = m_pBitmap->GetWidth(); //bitmap width
239  const UINT h = m_pBitmap->GetHeight(); //bitmap height
240 
241  for(UINT i=0; i<w; i++){
242  const float x = m_fOriginX + i/m_fScale; //noise X-coordinate
243 
244  for(UINT j=0; j<h; j++){
245  const float y = m_fOriginY + j/m_fScale; //noise Y-coordinate
246  const float noise = m_pPerlin->generate(x, y, t, m_nOctaves); //noise
247  SetPixel(i, j, noise); //draw noise pixel to bitmap
248 
249  //recompute maximum, minimum, and sum (for average)
250  m_fMin = min(m_fMin, noise);
251  m_fMax = max(m_fMax, noise);
252  m_fAve += noise;
253  } //for
254  } //for
255 
256  m_fAve /= w*h; //compute average from sum
257 
258  if(m_bShowGrid)DrawGrid();
259  if(m_bShowCoords)DrawCoords();
260 } //GenerateNoiseBitmap

267  const UINT nLeft = (UINT)floorf(rect.GetLeft() + point.X);

269  const UINT nTop = (UINT)floorf(rect.GetTop() + point.Y);
270  const UINT nBottom = (UINT)ceilf(rect.GetBottom() + point.Y);
271 
272  for(UINT i=nLeft; i<nRight; i++){
273  const float x = m_fOriginX + i/m_fScale;
274 
275  for(UINT j=nTop; j<nBottom; j++){
276  const float y = m_fOriginY + j/m_fScale;
277  SetPixel(i, j, m_pPerlin->generate(x, y, m_eNoise, m_nOctaves));
278  } //for
279  } //for
280 } //GenerateNoiseBitmap

305

333 void CMain::DrawGrid(){
334  const float fBitmapWidth = (float)m_pBitmap->GetWidth();
335  const float fBitmapHeight = (float)m_pBitmap->GetHeight();
336 
337  const UINT nv = (UINT)floorf(fBitmapWidth/m_fScale); //number of vertical lines
338  const UINT nh = (UINT)floorf(fBitmapHeight/m_fScale); //number of horizontal lines
339 
340  Gdiplus::Graphics graphics(m_pBitmap);
341  Gdiplus::Pen pen(Gdiplus::Color(255, 0, 255, 0)); //green
342 
343  //horizontal lines
344 
345  Gdiplus::PointF left(0.0f, m_fScale); //left X-coordinate of line
346  Gdiplus::PointF right(fBitmapWidth, m_fScale); //right X-coordinate of line
347  Gdiplus::RectF rect(0.0f, 0.0f, fBitmapWidth, 1.0f); //bounding rectangle
348 
349  for(UINT i=0; i<nh; i++){ //for each horizontal line
350  if(m_bShowGrid)graphics.DrawLine(&pen, left, right); //draw line
351  else GenerateNoiseBitmap(left, rect); //undraw line
352 
353  //move down to next horizontal line
354  left.Y += m_fScale;
355  right.Y += m_fScale;
356  } //for
357 
358  //vertical lines
359 
360  Gdiplus::PointF top(m_fScale, 0.0f); //top Y-coordinate of line
361  Gdiplus::PointF bottom(m_fScale, fBitmapHeight); //bottom Y-coordinate of line
362  rect = Gdiplus::RectF(0.0f, 0.0f, 1.0f, fBitmapHeight); //bounding rectangle
363 
364  for(UINT i=0; i<nv; i++){ //for each vertical line
365  if(m_bShowGrid)graphics.DrawLine(&pen, top, bottom); //draw line
366  else GenerateNoiseBitmap(top, rect); //undraw line
367 
368  //move right to next vertical line
369  top.X += m_fScale;
370  bottom.X += m_fScale;
371  } //for
372 
373  if(!m_bShowGrid && m_bShowCoords)
374  DrawCoords(); //in case we erased some of the coordinate text
375 } //DrawGrid
