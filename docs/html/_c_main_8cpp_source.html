68void CMain::OnPaint(){
69 PAINTSTRUCT ps; //paint structure
70 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
71 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
72
73 //get bitmap width and height
74
75 const int nBitmapWidth = m_pBitmap->GetWidth();
76 const int nBitmapHeight = m_pBitmap->GetHeight();
77
78 //get client rectangle
79
80 RECT rectClient; //for client rectangle
81 GetClientRect(m_hWnd, &rectClient); //get client rectangle
82 const int nClientWidth = rectClient.right - rectClient.left; //client width
83 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
84
85 //compute destination rectangle
86
87 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
88
89 const int width = min(nDestSide, nBitmapWidth); //dest rect width
90 const int height = min(nDestSide, nBitmapHeight); //dest rect height
91
92 const int x = max(0, nClientWidth - width)/2; //x margin
93 const int y = max(0, nClientHeight - height)/2; //y margin
94
95 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
96
97 //draw image to destination rectangle then clean up
98
99 graphics.DrawImage(m_pBitmap, rectDest);
100
101 EndPaint(m_hWnd, &ps); //this must be done last
102} //OnPaint

227void CMain::GenerateNoiseBitmap(eNoise t){
228 m_eNoise = t; //remember the noise type
229 UpdateMenus(); //changing noise type may change the menu status
230
231 for(UINT i=0; i<m_pBitmap->GetWidth(); i++){
232 const float x = m_fOriginX + i/m_fScale;
233
234 for(UINT j=0; j<m_pBitmap->GetHeight(); j++){
235 const float y = m_fOriginY + j/m_fScale;
236 SetPixel(i, j, m_pPerlin->generate(x, y, t, m_nOctaves));
237 } //for
238 } //for
239} //GenerateNoiseBitmap

385const std::wstring CMain::GetFileName() const{
386 std::wstring wstr;
387
388 switch(m_eNoise){
389 case eNoise::Perlin: wstr = L"Perlin"; break;
390 case eNoise::Value: wstr = L"Value"; break;
391 } //switch
392
393 switch(m_eHash){
394 case eHash::Permutation: wstr += L"-Perm"; break;
395 case eHash::Std: wstr += L"-Std"; break;
396 } //switch
397
398 switch(m_eDistr){
399 case eDistribution::Uniform: break; //nothing, which is the default
400 case eDistribution::Cosine: wstr += L"-Cos"; break;
401 case eDistribution::Normal: wstr += L"-Norm"; break;
402 case eDistribution::Exponential: wstr += L"-Exp"; break;
403 } //switch
404
405 switch(m_eSpline){
406 case eSpline::None: wstr += L"-NoSpline"; break;
407 case eSpline::Cubic: break; //nothing, which is the default
408 case eSpline::Quintic: wstr += L"-Quintic"; break;
409 } //switch
410
411 wstr += L"-" + std::to_wstring(m_nOctaves);
412 wstr += L"-" + std::to_wstring(1 << m_nLog2TableSize);
413 wstr += L"-" + std::to_wstring((size_t)round(m_fScale));
414
415 return wstr;
416} //GetFileName

421const std::wstring CMain::GetNoiseDescription() const{
422 std::wstring wstr;
423
424 if(m_eNoise == eNoise::Perlin || m_eNoise == eNoise::Value){
425 wstr += std::to_wstring(m_nOctaves) + L" octave";
426 if(m_nOctaves > 1)wstr += L"s";
427 wstr += L" of ";
428 } //if
429
430 switch(m_eNoise){
431 case eNoise::Perlin: wstr += L"Perlin"; break;
432 case eNoise::Value: wstr += L"Value"; break;
433 } //switch
434
435 wstr += L" Noise with origin (";
436 wstr += to_wstring_f(m_fOriginX, 2) + L", ";
437 wstr += to_wstring_f(m_fOriginY, 2) + L"), ";
438
439 switch(m_eHash){
440 case eHash::Permutation: wstr += L"a permutation"; break;
441 case eHash::Std: wstr += L"std"; break;
442 } //switch
443
444 wstr += L" hash function, ";
445
446 switch(m_eDistr){
447 case eDistribution::Uniform: wstr += L"uniform"; break;
448 case eDistribution::Cosine: wstr += L"cosine"; break;
449 case eDistribution::Normal: wstr += L"normal"; break;
450 case eDistribution::Exponential: wstr += L"exponential"; break;
451 } //switch
452
453 switch(m_eNoise){
454 case eNoise::Perlin: wstr += L" gradient"; break;
455 case eNoise::Value: wstr += L" height"; break;
456 } //switch
457
458 wstr += L" distribution, ";
459
460 switch(m_eSpline){
461 case eSpline::None: wstr += L"no"; break;
462 case eSpline::Cubic: wstr += L"cubic"; break; //nothing
463 case eSpline::Quintic: wstr += L"quintic"; break;
464 } //switch
465
466 wstr += L" spline function, scale ";
467 wstr += std::to_wstring((size_t)round(m_fScale));
468 wstr += L", and ";
469
470 if(m_eHash == eHash::Permutation)
471 wstr += L"permutation and ";
472
473 switch(m_eNoise){
474 case eNoise::Perlin: wstr += L"gradient "; break;
475 case eNoise::Value: wstr += L"height "; break;
476 } //switch
477
478 wstr += L"table size ";
479 wstr += std::to_wstring(1 << m_nLog2TableSize);
480 wstr += L". Lacunarity and persistence are fixed at 0.5 and 2.0, respectively.";
481
482 return wstr;
483} //GetNoiseDescription
