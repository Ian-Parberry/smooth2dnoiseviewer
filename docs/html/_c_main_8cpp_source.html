70void CMain::OnPaint(){
71 PAINTSTRUCT ps; //paint structure
72 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
73 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
74
75 //bitmap width and height
76
77 const int nBitmapWidth = m_pBitmap->GetWidth();
78 const int nBitmapHeight = m_pBitmap->GetHeight();
79
80 //get client rectangle
81
82 RECT rectClient; //for client rectangle
83 GetClientRect(m_hWnd, &rectClient); //get client rectangle
84 const int nClientWidth = rectClient.right - rectClient.left; //client width
85 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
86
87 //compute destination rectangle
88
89 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
90
91 const int width = min(nDestSide, nBitmapWidth); //dest rect width
92 const int height = min(nDestSide, nBitmapHeight); //dest rect height
93
94 const int x = max(0, nClientWidth - width)/2; //x margin
95 const int y = max(0, nClientHeight - height)/2; //y margin
96
97 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
98
99 //draw image to destination rectangle then clean up
100
101 graphics.DrawImage(m_pBitmap, rectDest);
102
103 EndPaint(m_hWnd, &ps); //this must be done last
104} //OnPaint

219void CMain::GeneratePixelNoise(){
220 m_eNoise = eNoise::Pixel;
221
222 EnableMenuItem(m_hFileMenu, IDM_FILE_SAVE, MF_ENABLED);
223 CheckMenuItem(m_hGenMenu, IDM_GENERATE_PIXELNOISE, MF_CHECKED);
224 CheckMenuItem(m_hGenMenu, IDM_GENERATE_PERLINNOISE, MF_UNCHECKED);
225 CheckMenuItem(m_hGenMenu, IDM_GENERATE_VALUENOISE, MF_UNCHECKED);
226 EnableMenuItem(m_hGenMenu, IDM_GENERATE_RANDOMIZE, MF_ENABLED);
227 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_NONE, MF_GRAYED);
228 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_CUBIC, MF_GRAYED);
229 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_QUINTIC, MF_GRAYED);
230
231 std::default_random_engine g;
232 g.seed(m_nSeed);
233
234 switch(m_eCurDist){
235 case eDistribution::Uniform: {
236 std::uniform_real_distribution<float> d(-1.0f, 1.0f);
237
238 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
239 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
240 SetPixel(i, j, d(g));
241 } //case
242 break;
243
244 case eDistribution::Cosine: {
245 std::uniform_real_distribution<float> d(0.0f, PI);
246
247 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
248 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
249 SetPixel(i, j, cosf(d(g)));
250 } //case
251 break;
252
253 case eDistribution::Normal:{
254 std::normal_distribution<float> d(500.0f, 200.0f);
255
256 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
257 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
258 SetPixel(i, j, 2.0f*clamp(0.0f, d(g)/1000.0f, 1.0f) - 1.0f);
259 } //case
260 break;
261
262 case eDistribution::Exponential: {
263 std::exponential_distribution<float> d(3.5f);
264
265 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
266 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
267 SetPixel(i, j, ((rand() & 1)? -1.0f: 1.0f)*clamp(0.0f, d(g), 1.0f));
268 } //case
269 break;
270 } //switch
271} //GeneratePixelNoise

276void CMain::GeneratePerlinNoise(eNoise t){
277 m_eNoise = t;
278
279 EnableMenuItem(m_hFileMenu, IDM_FILE_SAVE, MF_ENABLED);
280 CheckMenuItem(m_hGenMenu, IDM_GENERATE_PIXELNOISE, MF_UNCHECKED);
281 CheckMenuItem(m_hGenMenu, IDM_GENERATE_PERLINNOISE,
282 (t == eNoise::Perlin)? MF_CHECKED: MF_UNCHECKED);
283 CheckMenuItem(m_hGenMenu, IDM_GENERATE_VALUENOISE,
284 (t == eNoise::Value)? MF_CHECKED: MF_UNCHECKED);
285 EnableMenuItem(m_hGenMenu, IDM_GENERATE_RANDOMIZE, MF_ENABLED);
286 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_NONE, MF_ENABLED);
287 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_CUBIC, MF_ENABLED);
288 EnableMenuItem(m_hSplineMenu, IDM_SPLINE_QUINTIC, MF_ENABLED);
289
290 for(UINT i=0; i<m_pBitmap->GetWidth(); i++)
291 for(UINT j=0; j<m_pBitmap->GetHeight(); j++)
292 SetPixel(i, j, m_pPerlin->generate(i/m_fScale, j/m_fScale,
293 t, 0.5f, 2.0f, m_nNumOctaves));
294} //GeneratePerlinNoise

321void CMain::Regenerate(){
322 switch(m_eNoise){
323 case eNoise::Pixel:
324 GeneratePixelNoise();
325 break;
326
327 case eNoise::Perlin:
328 case eNoise::Value:
329 GeneratePerlinNoise(m_eNoise);
330 break;
331
332 default:
333 Clear();
334 break;
335 } //switch
336} //Regenerate

450const std::wstring CMain::GetFileName() const{
451 std::wstring wstr;
452
453 switch(m_eNoise){
454 case eNoise::Pixel: wstr = L"Pixel"; break;
455 case eNoise::Perlin: wstr = L"Perlin"; break;
456 case eNoise::Value: wstr = L"Value"; break;
457 } //switch
458
459 switch(m_eCurDist){
460 case eDistribution::Uniform: break; //nothing
461 case eDistribution::Cosine: wstr += L"-Cos"; break;
462 case eDistribution::Normal: wstr += L"-Norm"; break;
463 case eDistribution::Exponential: wstr += L"-Exp"; break;
464 } //switch
465
466 if(m_eNoise != eNoise::Pixel)
467 switch(m_eCurSpline){
468 case eSpline::None: L"-NoSpline"; break;
469 case eSpline::Cubic: break; //nothing
470 case eSpline::Quintic:wstr += L"-Quintic"; break;
471 } //switch
472
473 return wstr;
474} //GetFileName
