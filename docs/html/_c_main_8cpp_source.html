68void CMain::OnPaint(){
69 PAINTSTRUCT ps; //paint structure
70 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
71 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
72
73 //get bitmap width and height
74
75 const int nBitmapWidth = m_pBitmap->GetWidth();
76 const int nBitmapHeight = m_pBitmap->GetHeight();
77
78 //get client rectangle
79
80 RECT rectClient; //for client rectangle
81 GetClientRect(m_hWnd, &rectClient); //get client rectangle
82 const int nClientWidth = rectClient.right - rectClient.left; //client width
83 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
84
85 //compute destination rectangle
86
87 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
88
89 const int width = min(nDestSide, nBitmapWidth); //dest rect width
90 const int height = min(nDestSide, nBitmapHeight); //dest rect height
91
92 const int x = max(0, nClientWidth - width)/2; //x margin
93 const int y = max(0, nClientHeight - height)/2; //y margin
94
95 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
96
97 //draw image to destination rectangle then clean up
98
99 graphics.DrawImage(m_pBitmap, rectDest);
100
101 EndPaint(m_hWnd, &ps); //this must be done last
102} //OnPaint

213void CMain::GenerateNoiseBitmap(eNoise t){
214 m_eNoise = t; //remember the noise type
215 UpdateMenus(); //changing noise type may change the menu status
216
217 for(UINT i=0; i<m_pBitmap->GetWidth(); i++){
218 const float x = m_fOriginX + i/m_fScale;
219
220 for(UINT j=0; j<m_pBitmap->GetHeight(); j++){
221 const float y = m_fOriginY + j/m_fScale;
222 SetPixel(i, j, m_pPerlin->generate(x, y, t, m_nOctaves));
223 } //for
224 } //for
225} //GenerateNoiseBitmap

345const std::wstring CMain::GetFileName() const{
346 std::wstring wstr;
347
348 switch(m_eNoise){
349 case eNoise::Perlin: wstr = L"Perlin"; break;
350 case eNoise::Value: wstr = L"Value"; break;
351 } //switch
352
353 switch(m_eHash){
354 case eHash::Permutation: wstr += L"-Perm"; break;
355 case eHash::Std: wstr += L"-Std"; break;
356 } //switch
357
358 switch(m_eDistr){
359 case eDistribution::Uniform: break; //nothing, which is the default
360 case eDistribution::Cosine: wstr += L"-Cos"; break;
361 case eDistribution::Normal: wstr += L"-Norm"; break;
362 case eDistribution::Exponential: wstr += L"-Exp"; break;
363 } //switch
364
365 switch(m_eSpline){
366 case eSpline::None: wstr += L"-NoSpline"; break;
367 case eSpline::Cubic: break; //nothing, which is the default
368 case eSpline::Quintic: wstr += L"-Quintic"; break;
369 } //switch
370
371 wstr += L"-" + std::to_wstring(m_nOctaves);
372 wstr += L"-" + std::to_wstring(1 << m_nLog2TableSize);
373 wstr += L"-" + std::to_wstring((size_t)round(m_fScale));
374
375 return wstr;
376} //GetFileName

381const std::wstring CMain::GetNoiseDescription() const{
382 std::wstring wstr;
383
384 if(m_eNoise == eNoise::Perlin || m_eNoise == eNoise::Value){
385 wstr += std::to_wstring(m_nOctaves) + L" octave";
386 if(m_nOctaves > 1)wstr += L"s";
387 wstr += L" of ";
388 } //if
389
390 switch(m_eNoise){
391 case eNoise::Perlin: wstr += L"Perlin"; break;
392 case eNoise::Value: wstr += L"Value"; break;
393 } //switch
394
395 wstr += L" Noise with origin (";
396 wstr += to_wstring_f(m_fOriginX, 2) + L", ";
397 wstr += to_wstring_f(m_fOriginY, 2) + L"), ";
398
399 switch(m_eHash){
400 case eHash::Permutation: wstr += L"a permutation"; break;
401 case eHash::Std: wstr += L"std"; break;
402 } //switch
403
404 wstr += L" hash function, ";
405
406 switch(m_eDistr){
407 case eDistribution::Uniform: wstr += L"uniform"; break;
408 case eDistribution::Cosine: wstr += L"cosine"; break;
409 case eDistribution::Normal: wstr += L"normal"; break;
410 case eDistribution::Exponential: wstr += L"exponential"; break;
411 } //switch
412
413 switch(m_eNoise){
414 case eNoise::Perlin: wstr += L" gradient"; break;
415 case eNoise::Value: wstr += L" height"; break;
416 } //switch
417
418 wstr += L" distribution, ";
419
420 switch(m_eSpline){
421 case eSpline::None: wstr += L"no"; break;
422 case eSpline::Cubic: wstr += L"cubic"; break; //nothing
423 case eSpline::Quintic: wstr += L"quintic"; break;
424 } //switch
425
426 wstr += L" spline function, scale ";
427 wstr += std::to_wstring((size_t)round(m_fScale));
428 wstr += L", and ";
429
430 if(m_eHash == eHash::Permutation)
431 wstr += L"permutation and ";
432
433 switch(m_eNoise){
434 case eNoise::Perlin: wstr += L"gradient "; break;
435 case eNoise::Value: wstr += L"height "; break;
436 } //switch
437
438 wstr += L"table size ";
439 wstr += std::to_wstring(1 << m_nLog2TableSize);
440 wstr += L". Lacunarity and persistence are fixed at 0.5 and 2.0, respectively.";
441
442 return wstr;
443} //GetNoiseDescription
