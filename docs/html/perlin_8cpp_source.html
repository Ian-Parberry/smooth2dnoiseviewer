71void CPerlinNoise2D::Randomize(){
72 for(size_t i=0; i<m_nSize; i++) //identity permutation
73 m_nPerm[i] = i;
74
75 for(size_t i=0; i<m_nSize; i++) //randomize
76 std::swap(m_nPerm[i], m_nPerm[i + rand()%(m_nSize - i)]);
77} //Randomize

85void CPerlinNoise2D::Initialize(eDistribution d){
86 switch(d){
87 case eDistribution::Uniform:
88 for(size_t i=0; i<m_nSize; i++)
89 m_fTable[i] = (2.0f*i)/m_nSize - 1.0f;
90 break;
91
92 case eDistribution::Cosine:
93 for(size_t i=0; i<m_nSize; i++)
94 m_fTable[i] = cosf(((PI*i)/m_nSize));
95 break;
96
97 case eDistribution::Normal: {
98 std::default_random_engine g;
99 std::normal_distribution<float> d(500.0f, 200.0f);
100
101 for(size_t i=0; i<m_nSize; i++)
102 m_fTable[i] = 2.0f*clamp(0.0f, d(g)/1000.0f, 1.0f) - 1.0f;
103 } //case
104 break;
105
106 case eDistribution::Exponential: {
107 std::default_random_engine g;
108 std::exponential_distribution<float> d(3.5f);
109
110 const size_t half = m_nSize/2;
111
112 for(size_t i=0; i<half; i++) //positive values
113 m_fTable[i] = clamp(0.0f, d(g), 1.0f);
114
115 for(size_t i=half; i<m_nSize; i++) //negative values
116 m_fTable[i] = -clamp(0.0f, d(g), 1.0f);
117 } //case
118 break;
119 } //switch
120} //Initialize

163const float CPerlinNoise2D::noise(float x, float y, eNoise t) const{
164 const size_t nX = (size_t)floorf(x); //integer part of x
165 const size_t nY = (size_t)floorf(y); //integer part of y
166
167 const float fX = x - floorf(x); //fractional part of x
168 const float fY = y - floorf(y); //fractional part of y
169
170 //spline function
171 const float u = (m_eSpline == eSpline::Cubic)? spline3(fX): spline5(fX);
172 const float v = (m_eSpline == eSpline::Cubic)? spline3(fY): spline5(fY);
173
174 //hash value at corners of surrounding square with integer coordinates
175 const size_t AA = hash(pair(nX, nY));
176 const size_t BA = hash(pair(nX + 1, nY));
177 const size_t AB = hash(pair(nX, nY + 1));
178 const size_t BB = hash(pair(nX + 1, nY + 1));
179
180 //now lerp along the top and bottom in the x direction
181
182 float a=0, b=0; //for lerped values
183
184 switch(t){ //depending on whether Perlin noise or Value noise
185 case eNoise::Perlin:
186 a = lerp(u, grad(AA, fX, fY), grad(BA, fX - 1, fY));
187 b = lerp(u, grad(AB, fX, fY - 1), grad(BB, fX - 1, fY - 1));
188 break;
189
190 case eNoise::Value:
191 a = lerp(u, m_fTable[AA], m_fTable[BA]);
192 b = lerp(u, m_fTable[AB], m_fTable[BB]);
193 break;
194 } //switch
195
196 //now lerp in the other direction
197
198 return lerp(v, a, b);
199} //noise

211const float CPerlinNoise2D::generate(float x, float y, eNoise t, float alpha,
212 float beta, size_t n) const
213{
214 float sum = 0.0f, scale = 1.0f;
215
216 for(size_t i=0; i<n; i++){
217 sum += scale*noise(x, y, t);
218 scale *= alpha;
219 x *= beta;
220 y *= beta;
221 } //for
222
223 return (t == eNoise::Perlin? SQRT2: 1.0f)*(1.0f - alpha)*sum/(1.0f - scale);
224} //generate
