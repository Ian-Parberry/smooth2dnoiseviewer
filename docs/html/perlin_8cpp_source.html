89float CPerlinNoise2D::perlinnoise(float x, float y) {
90 const unsigned nX = (unsigned)floorf(x) & 255;
91 const unsigned nY = (unsigned)floorf(y) & 255;
92
93 const float fX = x - floorf(x);
94 const float fY = y - floorf(y);
95
96 const float u = spline3(fX);
97 const float v = spline3(fY);
98
99 const int AA = m_nPerm[m_nPerm[nX] + nY];
100 const int BA = m_nPerm[m_nPerm[nX + 1] + nY];
101 const int AB = m_nPerm[m_nPerm[nX] + nY + 1];
102 const int BB = m_nPerm[m_nPerm[nX + 1] + nY + 1];
103
104 const float a = lerp(u, grad(AA, fX, fY), grad(BA, fX - 1, fY));
105 const float b = lerp(u, grad(AB, fX, fY - 1), grad(BB, fX - 1, fY - 1));
106
107 return lerp(v, a, b);
108} //perlinnoise
