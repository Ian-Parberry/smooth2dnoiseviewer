83 m_nSeed = timeGetTime();
84
85 for(size_t i=0; i<m_nSize; i++) //identity permutation
86 m_nPerm[i] = i;

109void CPerlinNoise2D::MidpointDisplacement(size_t i, size_t j, float alpha){
110 std::uniform_real_distribution<float> d(-1.0f, 1.0f);
111
112 if(j > i + 1){ //base of recursion is "do nothing"
113 const size_t mid = (i + j)/2; //mid point
114
115 const float fMean = (m_fTable[i] + m_fTable[j - 1])/2.0f; //average of ends
116 const float fRand = d(*m_pRandom); //random value in [-1, 1]
117 m_fTable[mid] = clamp(-1.0f, fMean + fRand, 1.0f); //mid point is average plus offset
118 alpha *= 0.5f; //increase lacunarity
119
120 MidpointDisplacement(i, mid, alpha); //recurse on first half
121 MidpointDisplacement(mid, j, alpha); //recurse on second half
122 } //if
123} //MidpointDisplacement

132void CPerlinNoise2D::RandomizeTable(eDistribution d){
133 switch(d){
134 case eDistribution::Uniform: {
135 std::uniform_real_distribution<float> d(-1.0f, 1.0f);
136
137 for(size_t i=0; i<m_nSize; i++)
138 m_fTable[i] = d(*m_pRandom);
139 } //case
140 break;
141
142 case eDistribution::Cosine: {
143 std::uniform_real_distribution<float> d(0.0f, 1.0f);
144
145 for(size_t i=0; i<m_nSize; i++)
146 m_fTable[i] = cosf(PI*d(*m_pRandom));
147 } //case
148 break;
149
150 case eDistribution::Normal: {
151 std::normal_distribution<float> d(500.0f, 200.0f);
152
153 for(size_t i=0; i<m_nSize; i++)
154 m_fTable[i] = 2.0f*clamp(0.0f, d(*m_pRandom)/1000.0f, 1.0f) - 1.0f;
155 } //case
156 break;
157
158 case eDistribution::Exponential: {
159 std::exponential_distribution<float> d(8.0f);
160
161 const size_t half = m_nSize/2;
162
163 for(size_t i=0; i<half; i++) //positive values
164 m_fTable[i] = clamp(0.0f, d(*m_pRandom), 1.0f);
165
166 for(size_t i=half; i<m_nSize; i++) //negative values
167 m_fTable[i] = -clamp(0.0f, d(*m_pRandom), 1.0f);
168 } //case
169 break;
170
171 case eDistribution::MidpointDisplacement:
172 m_fTable[0] = 1.0f;
173 m_fTable[m_nSize - 1] = -1.0f;
174
175 MidpointDisplacement(0, m_nSize, 0.5f);
176 break;
177 } //switch
178} //RandomizeTable

224inline const float CPerlinNoise2D::grad(size_t h, float x, float y) const{
225 h = h & m_nMask; //safety
226 return clamp(-1.0f, x*m_fTable[h] + y*m_fTable[hash(h)], 1.0f);

269void CPerlinNoise2D::HashCorners(size_t x, size_t y, size_t c[4]) const{
270 switch(m_eHash){
271 case eHash::Permutation:
272 c[0] = hash(pair(x, y)); c[1] = hash(pair(x + 1, y));
273 c[2] = hash(pair(x, y + 1)); c[3] = hash(pair(x + 1, y + 1));
274 break;
275
276 case eHash::Std:
277 c[0] = hashstd(pairstd(x, y)); c[1] = hashstd(pairstd(x + 1, y));
278 c[2] = hashstd(pairstd(x, y + 1)); c[3] = hashstd(pairstd(x + 1, y + 1));
279 break;
280 } //switch
281} //HashCorners

296const float CPerlinNoise2D::noise(float x, float y, eNoise t) const{
297 const size_t nX = (size_t)floorf(x); //integer part of x
298 const size_t nY = (size_t)floorf(y); //integer part of y
299
300 const float fX = x - floorf(x); //fractional part of x
301 const float fY = y - floorf(y); //fractional part of y
302
303 //smooth fractional parts of x and y using spline curves
304
305 const float sX = spline(fX); //apply spline curve to fractional part of x
306 const float sY = spline(fY); //apply spline curve to fractional part of y
307
308 //hash value at corners of enclosing grid square with integer coordinates
309
310 size_t c[4] = {0}; //for hashed values at corners
311 HashCorners(nX, nY, c); //get hashed values at corners
312
313 //lerp along the top and bottom in the x direction
314
315 float a=0, b=0; //for lerped values
316
317 switch(t){ //depending on whether Perlin noise or Value noise
318 case eNoise::Perlin: //lerp gradients
319 a = lerp(sX, grad(c[0], fX, fY), grad(c[1], fX - 1, fY));
320 b = lerp(sX, grad(c[2], fX, fY - 1), grad(c[3], fX - 1, fY - 1));
321 break;
322
323 case eNoise::Value: //lerp heights
324 a = lerp(sX, m_fTable[c[0]], m_fTable[c[1]]);
325 b = lerp(sX, m_fTable[c[2]], m_fTable[c[3]]);
326 break;
327 } //switch
328
329 //now lerp in the other direction
330
331 return lerp(sY, a, b);
332} //noise

347const float CPerlinNoise2D::generate(float x, float y, eNoise t, size_t n,
348 float alpha, float beta) const
349{
350 float sum = 0.0f; //for result
351 float amplitude = 1.0f;
352
353 for(size_t i=0; i<n; i++){ //for each octave
354 sum += amplitude*noise(x, y, t); //scale noise by amplitude
355 amplitude *= alpha; //reduce amplitude by lacunarity
356 x *= beta; y *= beta; //multiply frequency by persistence
357 } //for
358
359 //scale result by sum of geometric progression
360 const float result = (1 - alpha)*sum/(1 - amplitude);
361
362 //scale up for Perlin noise
363 return (t == eNoise::Perlin)? SQRT2*result: result;
364} //generate
