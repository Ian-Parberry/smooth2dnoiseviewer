85 void CPerlinNoise2D::Initialize(eDistribution d){
86  switch(d){
87  case eDistribution::Uniform:
88  for(size_t i=0; i<m_nSize; i++)
89  m_fTable[i] = (2.0f*i)/m_nSize - 1.0f;
90  break;
91 
92  case eDistribution::Cosine:
93  for(size_t i=0; i<m_nSize; i++)
94  m_fTable[i] = cosf(((PI*i)/m_nSize));
95  break;
96 
97  case eDistribution::Normal: {
98  std::default_random_engine g;
99  std::normal_distribution<float> d(500.0f, 200.0f);
100 
101  for(size_t i=0; i<m_nSize; i++)
102  m_fTable[i] = 2.0f*clamp(0.0f, d(g)/1000.0f, 1.0f) - 1.0f;
103  } //case
104  break;
105 
106  case eDistribution::Exponential: {
107  std::default_random_engine g;
108  std::exponential_distribution<float> d(3.5f);
109 
110  const size_t half = m_nSize/2;
111 
112  for(size_t i=0; i<half; i++) //positive values
113  m_fTable[i] = clamp(0.0f, d(g), 1.0f);
114 
115  for(size_t i=half; i<m_nSize; i++) //negative values
116  m_fTable[i] = -clamp(0.0f, d(g), 1.0f);
117  } //case
118  break;
119  } //switch
120 } //Initialize

163 const float CPerlinNoise2D::noise(float x, float y, eNoise t) const{
164  const size_t nX = (size_t)floorf(x); //integer part of x
165  const size_t nY = (size_t)floorf(y); //integer part of y
166 
167  const float fX = x - floorf(x); //fractional part of x
168  const float fY = y - floorf(y); //fractional part of y
169 
170  float u, v; //for spline function along x-axis
171 
172  switch(m_eSpline){
173  case eSpline::None: u = fX; v = fY; break;
174  case eSpline::Cubic: u = spline3(fX); v = spline3(fY); break;
175  case eSpline::Quintic: u = spline5(fX); v = spline5(fY);break;
176  } //switch
177 
178  //hash value at corners of surrounding square with integer coordinates
179 
180  const size_t AA = hash(pair(nX, nY));
181  const size_t BA = hash(pair(nX + 1, nY));
182  const size_t AB = hash(pair(nX, nY + 1));
183  const size_t BB = hash(pair(nX + 1, nY + 1));
184 
185  //now lerp along the top and bottom in the x direction
186 
187  float a=0, b=0; //for lerped values
188 
189  switch(t){ //depending on whether Perlin noise or Value noise
190  case eNoise::Perlin:
191  a = lerp(u, grad(AA, fX, fY), grad(BA, fX - 1, fY));
192  b = lerp(u, grad(AB, fX, fY - 1), grad(BB, fX - 1, fY - 1));
193  break;
194 
195  case eNoise::Value:
196  a = lerp(u, m_fTable[AA], m_fTable[BA]);
197  b = lerp(u, m_fTable[AB], m_fTable[BB]);
198  break;
199  } //switch
200 
201  //now lerp in the other direction
202 
203  return lerp(v, a, b);
204 } //noise

216 const float CPerlinNoise2D::generate(float x, float y, eNoise t, float alpha,
217  float beta, size_t n) const
218 {
219  float sum = 0.0f; //for sum of octaves
220  float scale = 1.0f; //for octave scale
221 
222  for(size_t i=0; i<n; i++){ //for each octave
223  sum += scale*noise(x, y, t); //scale noise by octave scale
224  scale *= alpha; //reduce octave scale
225  x *= beta;
226  y *= beta;
227  } //for
228 
229  //scale result by sum of geometric progression
230  const float result = (1 - alpha)*sum/(1 - scale);
231 
232  //scale up for Perlin noise
233  return (t == eNoise::Perlin)? SQRT2*result: result;
234 } //generate
