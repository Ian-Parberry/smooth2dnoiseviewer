80void CPerlinNoise2D::RandomizePermutation(){
81 m_nSeed = timeGetTime();
82
83 for(size_t i=0; i<m_nSize; i++) //identity permutation
84 m_nPerm[i] = i;
85
86 for(size_t i=0; i<m_nSize; i++) //randomize
87 std::swap(m_nPerm[i], m_nPerm[i + rand()%(m_nSize - i)]);
88} //RandomizePermutation

107void CPerlinNoise2D::RandomizeTableMidpoint(size_t i, size_t j, float alpha){
108 std::uniform_real_distribution<float> d(-1.0f, 1.0f);
109
110 if(j > i + 1){ //base of recursion is "do nothing"
111 const size_t mid = (i + j)/2; //mid point
112
113 const float fMean = (m_fTable[i] + m_fTable[j - 1])/2.0f; //average of ends
114 const float fRand = d(m_stdRandom); //random value in [-1, 1]
115 m_fTable[mid] = clamp(-1.0f, fMean + fRand, 1.0f); //mid point is average plus offset
116 alpha *= 0.5f; //increase lacunarity
117
118 RandomizeTableMidpoint(i, mid, alpha); //recurse on first half
119 RandomizeTableMidpoint(mid, j, alpha); //recurse on second half
120 } //if
121} //RandomizeTableMidpoint

161void CPerlinNoise2D::RandomizeTableExp(){
162 std::exponential_distribution<float> d(8.0f);
163
164 const size_t half = m_nSize/2;
165
166 for(size_t i=0; i<half; i++) //positive values
167 m_fTable[i] = clamp(0.0f, d(m_stdRandom), 1.0f);
168
169 for(size_t i=half; i<m_nSize; i++) //negative values
170 m_fTable[i] = -clamp(0.0f, d(m_stdRandom), 1.0f);
171} //RandomizeTableExp

239inline const float CPerlinNoise2D::z(size_t h, float x, float y, eNoise t) const{
240 h = h & m_nMask; //safety
241 float result = 0; //return result
242
243 switch(t){ //noise type
244 case eNoise::Perlin:
245 result = x*m_fTable[h] + y*m_fTable[hash(h)]; //gradient times position
246 break;
247
248 case eNoise::Value:
249 result = m_fTable[h]; //get height directly from table
250 break;
251 } //switch
252
253 return clamp(-1.0f, result, 1.0f); //safety
254} //z

301void CPerlinNoise2D::HashCorners(size_t x, size_t y, size_t c[4]) const{
302 switch(m_eHash){
303 case eHash::Permutation:
304 c[0] = hash(pair(x, y)); c[1] = hash(pair(x + 1, y));
305 c[2] = hash(pair(x, y + 1)); c[3] = hash(pair(x + 1, y + 1));
306 break;
307
308 case eHash::Std:
309 c[0] = hashstd(pairstd(x, y)); c[1] = hashstd(pairstd(x + 1, y));
310 c[2] = hashstd(pairstd(x, y + 1)); c[3] = hashstd(pairstd(x + 1, y + 1));
311 break;
312 } //switch
313} //HashCorners

343const float CPerlinNoise2D::noise(float x, float y, eNoise t) const{
344 const size_t nX = (size_t)floorf(x); //integer part of x
345 const size_t nY = (size_t)floorf(y); //integer part of y
346
347 const float fX = x - floorf(x); //fractional part of x
348 const float fY = y - floorf(y); //fractional part of y
349
350 //smooth fractional parts of x and y using spline curves
351
352 const float sX = spline(fX); //apply spline curve to fractional part of x
353 const float sY = spline(fY); //apply spline curve to fractional part of y
354
355 //hash value at corners of enclosing grid square with integer coordinates
356
357 size_t c[4] = {0}; //for hashed values at corners
358 HashCorners(nX, nY, c); //get hashed values at corners
359
360 //lerp along the top and bottom along the X-axis
361
362 const float a = Lerp(sX, fX, fY, c, t);
363 const float b = Lerp(sX, fX, fY - 1, &(c[2]), t);
364
365 //now lerp these values along the Y-axis
366
367 return lerp(sY, a, b);
368} //noise

383const float CPerlinNoise2D::generate(float x, float y, eNoise t, size_t n,
384 float alpha, float beta) const
385{
386 float sum = 0.0f; //for result
387 float amplitude = 1.0f;
388
389 for(size_t i=0; i<n; i++){ //for each octave
390 sum += amplitude*noise(x, y, t); //scale noise by amplitude
391 amplitude *= alpha; //reduce amplitude by lacunarity
392 x *= beta; y *= beta; //multiply frequency by persistence
393 } //for
394
395 return (1 - alpha)*sum/(1 - amplitude); //scale by sum of geometric progression
396} //generate
