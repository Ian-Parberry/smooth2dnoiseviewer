91 void CPerlinNoise2D::RandomizePermutation(){
92  for(size_t i=0; i<m_nSize; i++) //identity permutation
93  m_nPerm[i] = i;
94 
95  m_stdRandom.seed(m_nSeed); //reset PRNG
96 
97  for(size_t i=0; i<m_nSize; i++){ //randomize
98  std::uniform_int_distribution<size_t> d(i, m_nSize - 1);
99  std::swap(m_nPerm[i], m_nPerm[d(m_stdRandom)]);
100  } //for

121 void CPerlinNoise2D::RandomizeTableMidpoint(size_t i, size_t j, float alpha){
122  assert(i < j && j < m_nSize);
123  assert(alpha < 0.0f);
124 
125  std::uniform_real_distribution<float> d(-1.0f, 1.0f);
126 
127  if(j > i + 1){ //there is a midpoint to fill in
128  const size_t mid = (i + j)/2; //mid point
129 
130  assert(i < mid && mid < j);
131 
132  const float fMean = (m_fTable[i] + m_fTable[j])/2.0f; //average of ends
133  const float fRand = alpha*d(m_stdRandom); //random offset
134  m_fTable[mid] = clamp(-1.0f, fMean + fRand, 1.0f); //mid point is average plus offset
135  alpha *= 0.5f; //increase lacunarity
136 
137  RandomizeTableMidpoint(i, mid, alpha); //recurse on first half
138  RandomizeTableMidpoint(mid, j, alpha); //recurse on second half
139  } //if
140 } //RandomizeTableMidpoint

407 void CPerlinNoise2D::HashCorners(size_t x, size_t y, size_t c[4]) const{
408  switch(m_eHash){
409  case eHash::Permutation:
410  c[0] = hash(pair(x, y)); c[1] = hash(pair(x + 1, y));
411  c[2] = hash(pair(x, y + 1)); c[3] = hash(pair(x + 1, y + 1));
412  break;
413 
414  case eHash::LinearCongruential:
415  c[0] = hash2(x, y); c[1] = hash2(x + 1, y);
416  c[2] = hash2(x, y + 1); c[3] = hash2(x + 1, y + 1);
417  break;
418 
419  case eHash::Std:
420  c[0] = hashstd(pairstd(x, y)); c[1] = hashstd(pairstd(x + 1, y));
421  c[2] = hashstd(pairstd(x, y + 1)); c[3] = hashstd(pairstd(x + 1, y + 1));
422  break;
423  } //switch
424 } //HashCorners

438  assert(-1.0f <= x && x <= 1.0f);

483  return result;
484  break;
485

487  assert(-1.0f <= result && result <= 1.0f);

508 const float CPerlinNoise2D::noise(float x, float y, eNoise t) const{
509  const size_t nX = (size_t)floorf(x); //integer part of x
510  const size_t nY = (size_t)floorf(y); //integer part of y
511 
512  const float fX = x - floorf(x); //fractional part of x
513  const float fY = y - floorf(y); //fractional part of y
514 
515  //smooth fractional parts of x and y using spline curves
516 
517  const float sX = spline(fX); //apply spline curve to fractional part of x
518  const float sY = spline(fY); //apply spline curve to fractional part of y
519 
520  //hash value at corners of enclosing grid square with integer coordinates
521 
522  size_t c[4] = {0}; //for hashed values at corners
523  HashCorners(nX, nY, c); //get hashed values at corners
524 
525  //lerp along the top and bottom along the X-axis
526 
527  const float a = Lerp(sX, fX, fY, c, t);
528  const float b = Lerp(sX, fX, fY - 1, &(c[2]), t);
529 
530  //now lerp these values along the Y-axis
531 
532  const float result = lerp(sY, a, b);
533  assert(-1.0f <= result && result <= 1.0f);
534  return result;
535 } //noise

550 const float CPerlinNoise2D::generate(float x, float y, eNoise t, size_t n,
551  float alpha, float beta) const
552 {
553  assert(0.0f <= alpha && alpha < 1.0f);
554  assert(beta > 1.0f);
555 
556  float sum = 0.0f; //for result
557  float amplitude = 1.0f; //octave amplitude
558 
559  for(size_t i=0; i<n; i++){ //for each octave
560  sum += amplitude*noise(x, y, t); //scale noise by amplitude
561  amplitude *= alpha; //reduce amplitude by lacunarity
562  x *= beta; y *= beta; //multiply frequency by persistence
563  } //for
564 
565  assert(amplitude == powf(alpha, (float)n));
566 
567  float result = (1 - alpha)*sum/(1 - amplitude); //sum of geometric progression
568  if(t == eNoise::Perlin)result *= 4.0f/3.0f; //scale up Perlin noise
569  assert(-1.0f <= result && result <= 1.0f); //safety
570  return result;
571 } //generate
