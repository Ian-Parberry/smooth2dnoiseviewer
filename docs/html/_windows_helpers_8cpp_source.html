62HRESULT GetEncoderClsid(const WCHAR* format, CLSID* pClsid){
63 UINT num = 0; //number of image encoders
64 UINT n = 0; //size of the image encoder array in bytes
65 HRESULT hr = E_FAIL; //return result
66
67 Gdiplus::ImageCodecInfo* pCodecInfo = nullptr; //for codec info
68 if(FAILED(Gdiplus::GetImageEncodersSize(&num, &n)))return E_FAIL; //get sizes
69 if(n == 0)return E_FAIL; //there are no encoders
70
71 pCodecInfo = (Gdiplus::ImageCodecInfo*)(malloc(n)); //allocate codec info memory
72 if(pCodecInfo == nullptr)return E_FAIL; //malloc failed (as if)
73 if(FAILED(GetImageEncoders(num, n, pCodecInfo)))return E_FAIL; //get encoders
74
75 for(UINT j=0; j<num && hr!=S_OK; j++) //for each encoder, while not found
76 if(wcscmp(pCodecInfo[j].MimeType, format) == 0){ //found the codex we want
77 *pClsid = pCodecInfo[j].Clsid; //return it
78 hr = S_OK; //success
79 } //if
80
81 free(pCodecInfo); //clean up
82 return hr;
83} //GetEncoderClsid

96HRESULT SaveBitmap(HWND hwnd, Gdiplus::Bitmap* pBitmap){
97 COMDLG_FILTERSPEC filetypes[] = { //png files only
98 {L"PNG Files", L"*.png"}
99 }; //filetypes
100
101 std::wstring wstrFileName; //result
102 CComPtr<IFileSaveDialog> pDlg; //pointer to save dialog box
103 static int n = 0; //number of images saved in this run
104 std::wstring wstrName = L"Image" + std::to_wstring(n++); //default file name
105 CComPtr<IShellItem> pItem; //item pointer
106 LPWSTR pwsz = nullptr; //pointer to null-terminated wide string for result
107
108 //fire up the save dialog box
109
110 if(FAILED(pDlg.CoCreateInstance(__uuidof(FileSaveDialog))))return E_FAIL;
111
112 pDlg->SetFileTypes(_countof(filetypes), filetypes); //set file types to png
113 pDlg->SetTitle(L"Save Image"); //set title bar text
114 pDlg->SetFileName(wstrName.c_str()); //set default file name
115 pDlg->SetDefaultExtension(L"png"); //set default extension
116
117 if(FAILED(pDlg->Show(hwnd)))return E_FAIL; //show the dialog box
118 if(FAILED(pDlg->GetResult(&pItem)))return E_FAIL; //get the result item
119 if(FAILED(pItem->GetDisplayName(SIGDN_FILESYSPATH, &pwsz)))return E_FAIL; //get file name
120
121 //wstrFileName should now contain the selected file name
122
123 CLSID clsid; //for PNG class id
124 if(FAILED(GetEncoderClsid((WCHAR*)L"image/png", &clsid)))return E_FAIL; //get
125 pBitmap->Save(pwsz, &clsid, nullptr); //the actual save happens here
126
127 CoTaskMemFree(pwsz); //clean up
128
129 return S_OK;
130} //SaveBitmap
